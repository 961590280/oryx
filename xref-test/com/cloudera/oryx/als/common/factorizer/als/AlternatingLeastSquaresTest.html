37    @Test
38    public void testALS() throws Exception {
39      RealMatrix product = buildTestXYTProduct();
40  
41      assertArrayEquals(
42          new float[] {-0.030258f, 0.852781f, 1.004839f, 1.024087f, -0.036206f},
43          product.getRow(0));
44      assertArrayEquals(
45          new float[] {0.077046f, 0.751232f, 0.949796f, 0.910322f, 0.073047f},
46          product.getRow(1));
47      assertArrayEquals(
48          new float[] {0.916777f, -0.196005f, 0.335926f, -0.163591f, 0.929028f},
49          product.getRow(2));
50      assertArrayEquals(
51          new float[] {0.987400f, 0.130943f, 0.772403f, 0.235522f, 0.998354f},
52          product.getRow(3));
53      assertArrayEquals(
54          new float[] {-0.028683f, 0.850540f, 1.003130f, 1.021514f, -0.034598f},
55          product.getRow(4));
56    }
57  
58    static RealMatrix buildTestXYTProduct() throws ExecutionException, InterruptedException {
59  
60      LongObjectMap<LongFloatMap> byRow = new LongObjectMap<>();
61      LongObjectMap<LongFloatMap> byCol = new LongObjectMap<>();
62      // Octave: R = [ 0 2 3 1 0 ; 0 0 4 5 0 ; 1 0 0 0 2 ; 3 0 1 0 5 ; 0 2 2 2 0 ]
63      MatrixUtils.addTo(0, 1, 2.0f, byRow, byCol);
64      MatrixUtils.addTo(0, 2,  3.0f, byRow, byCol);
65      MatrixUtils.addTo(0, 3,  1.0f, byRow, byCol);
66      MatrixUtils.addTo(1, 2,  4.0f, byRow, byCol);
67      MatrixUtils.addTo(1, 3,  5.0f, byRow, byCol);
68      MatrixUtils.addTo(2, 0,  1.0f, byRow, byCol);
69      MatrixUtils.addTo(2, 4,  2.0f, byRow, byCol);
70      MatrixUtils.addTo(3, 0,  3.0f, byRow, byCol);
71      MatrixUtils.addTo(3, 2,  1.0f, byRow, byCol);
72      MatrixUtils.addTo(3, 4,  5.0f, byRow, byCol);
73      MatrixUtils.addTo(4, 1,  2.0f, byRow, byCol);
74      MatrixUtils.addTo(4, 2,  2.0f, byRow, byCol);
75      MatrixUtils.addTo(4, 3,  2.0f, byRow, byCol);
76  
77      // Octave: Y = [ 0.1 0.2 ; 0.2 0.5 ; 0.3 0.1 ; 0.2 0.2 ; 0.5 0.4 ];
78      LongObjectMap<float[]> previousY = new LongObjectMap<>();
79      previousY.put(0L, new float[] {0.1f, 0.2f});
80      previousY.put(1L, new float[] {0.2f, 0.5f});
81      previousY.put(2L, new float[] {0.3f, 0.1f});
82      previousY.put(3L, new float[] {0.2f, 0.2f});
83      previousY.put(4L, new float[] {0.5f, 0.4f});
84  
85      MatrixFactorizer als = new AlternatingLeastSquares(byRow, byCol, 2, 0.0001, 40);
86      als.setPreviousY(previousY);
87      als.call();
88  
89      return MatrixUtilsTest.multiplyXYT(als.getX(), als.getY());
90    }
