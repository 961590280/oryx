28  /**
29   * Tests {@link WeightedPrediction}.

31   * @author Sean Owen
32   */
33  public final class WeightedPredictionTest extends OryxTest {
34  
35    @Test
36    public void testNumericVote() {
37      List<NumericPrediction> predictions = Arrays.asList(
38          new NumericPrediction(1.0f, 1),
39          new NumericPrediction(3.0f, 2),
40          new NumericPrediction(6.0f, 3)
41      );
42      double[] weights = {1.0, 1.0, 1.0};
43      NumericPrediction vote = (NumericPrediction) WeightedPrediction.voteOnFeature(predictions, weights);
44      assertEquals(FeatureType.NUMERIC, vote.getFeatureType());
45      assertEquals(10.0f/3.0f, vote.getPrediction());
46    }
47  
48    @Test
49    public void testNumericVoteWeighted() {
50      List<NumericPrediction> predictions = Arrays.asList(
51          new NumericPrediction(1.0f, 1),
52          new NumericPrediction(3.0f, 2),
53          new NumericPrediction(6.0f, 3)
54      );
55      double[] weights = {3.0, 2.0, 1.0};
56      NumericPrediction vote = (NumericPrediction) WeightedPrediction.voteOnFeature(predictions, weights);
57      assertEquals(FeatureType.NUMERIC, vote.getFeatureType());
58      assertEquals(15.0f/6.0f, vote.getPrediction());
59    }
60  
61    @Test
62    public void testCategoricalVote() {
63      List<CategoricalPrediction> predictions = Arrays.asList(
64          new CategoricalPrediction(new int[] {0, 1, 2}),
65          new CategoricalPrediction(new int[] {6, 2, 0}),
66          new CategoricalPrediction(new int[] {0, 2, 0})
67      );
68      double[] weights = {1.0, 1.0, 1.0};
69      CategoricalPrediction vote = (CategoricalPrediction) WeightedPrediction.voteOnFeature(predictions, weights);
70      assertEquals(FeatureType.CATEGORICAL, vote.getFeatureType());
71      assertEquals(1, vote.getMostProbableCategoryID());
72    }
73  
74    @Test
75    public void testCategoricalVoteWeighted() {
76      List<CategoricalPrediction> predictions = Arrays.asList(
77          new CategoricalPrediction(new int[] {0, 1, 2}),
78          new CategoricalPrediction(new int[] {6, 2, 0}),
79          new CategoricalPrediction(new int[] {0, 2, 0})
80      );
81      double[] weights = {1.0, 10.0, 1.0};
82      CategoricalPrediction vote = (CategoricalPrediction) WeightedPrediction.voteOnFeature(predictions, weights);
83      assertEquals(FeatureType.CATEGORICAL, vote.getFeatureType());
84      assertEquals(0, vote.getMostProbableCategoryID());
85    }
