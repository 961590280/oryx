37  /**
38   * <p>Tests the random decision forest classifier on a
39   * <a href="http://archive.ics.uci.edu/ml/datasets/Wine+Quality">wine quality</a> data set. This contains
40   * a numeric features and a numeric target.</p>
41   *
42   * <p>Citation for data set: P. Cortez, A. Cerdeira, F. Almeida, T. Matos and J. Reis. Modeling wine 
43   * preferences by data mining from physicochemical properties. In Decision Support Systems, Elsevier,
44   * 47(4):547-553, 2009.</p>
45   *
46   * @author Sean Owen
47   */
48  public final class WineQualityIT extends AbstractComputationIT {
49  
50    private static final Logger log = LoggerFactory.getLogger(WineQualityIT.class);
51  
52    @Override
53    protected File getTestDataPath() {
54      return getResourceAsFile("winequality");
55    }
56  
57    @Test
58    public void testWineQuality() throws Exception {
59      List<Example> allExamples = readWineQualityExamples();
60      DecisionForest forest = DecisionForest.fromExamplesWithDefault(allExamples);
61      log.info("Evals: {}", forest.getEvaluations());
62      assertTrue(new Mean().evaluate(forest.getEvaluations()) < 1.2);
63      double[] importances = forest.getFeatureImportances();
64      log.info("Importances: {}", importances);
65      for (double d : importances) {
66        assertTrue(d >= 0.0);
67        assertTrue(d <= 1.0);
68      }
69      // Just too flaky:
70      /*
71      double mostImportant = Doubles.max(importances);
72      assertTrue(importances[8] == mostImportant || importances[10] == mostImportant);
73      assertTrue(importances[8] > 0.7);
74      assertTrue(importances[10] > 0.55);
75       */
76    }
77  
78    @Test
79    public void testPMMLOutput() throws Exception {
80      new RDFLocalGenerationRunner().call();
81      File pmmlFile = new File(TEST_TEMP_BASE_DIR, "00000/model.pmml.gz");
82      DecisionForestPMML.read(pmmlFile);
83    }
84  
85    private static List<Example> readWineQualityExamples() throws IOException {
86      List<Example> allExamples = new ArrayList<>();
87      Pattern delimiter = Pattern.compile(";");
88      File dataFile = new File(TEST_TEMP_INBOUND_DIR, "winequality-white.csv");
89      for (CharSequence line : new FileLineIterable(dataFile)) {
90        if (line.length() == 0) {
91          continue;
92        }
93        String[] tokens = delimiter.split(line);
94        Feature[] features = new Feature[11];
95        for (int i = 0; i < features.length; i++) {
96          features[i] = NumericFeature.forValue(Float.parseFloat(tokens[i]));
97        }
98        Example trainingExample = new Example(NumericFeature.forValue(Float.parseFloat(tokens[11])), features);
99        allExamples.add(trainingExample);
100     }
101     return allExamples;
102   }
