43  /**
44   * Tests the random decision forest classifier on the classic
45   * <a href="http://archive.ics.uci.edu/ml/datasets/Iris">Iris data set</a>. This contains
46   * a few numeric features and a categorical target.
47   *
48   * @author Sean Owen
49   */
50  public final class IrisIT extends AbstractComputationIT {
51  
52    private static final Logger log = LoggerFactory.getLogger(IrisIT.class);
53  
54    @Override
55    protected File getTestDataPath() {
56      return getResourceAsFile("iris");
57    }
58  
59    @Test
60    public void testPMMLOutput() throws Exception {
61      // It's not clear this will actually be deterministic but it will probably be for our purposes
62      new RDFLocalGenerationRunner().call();
63      File pmmlFile = new File(TEST_TEMP_BASE_DIR, "00000/model.pmml.gz");
64  
65      try (Reader pmmlReader = IOUtils.openReaderMaybeDecompressing(pmmlFile)) {
66        log.info("PMML:\n{}", CharStreams.toString(pmmlReader));
67      }
68  
69      Pair<DecisionForest,Map<Integer,BiMap<String,Integer>>> forestAndMapping = DecisionForestPMML.read(pmmlFile);
70      DecisionForest forest = forestAndMapping.getFirst();
71      Map<Integer,BiMap<String,Integer>> categoryValueMapping = forestAndMapping.getSecond();
72      Map<String,Integer> targetCategoryValueMapping = categoryValueMapping.get(4);
73  
74      log.info("{}", forest);
75  
76      double[] importances = forest.getFeatureImportances();
77      for (double d : importances) {
78        assertTrue(d >= 0.0);
79        assertTrue(d <= 1.0);
80      }
81      // petal length ought to be most predictive
82      InboundSettings settings = InboundSettings.create(ConfigUtils.getDefaultConfig());
83      int petalLengthColumn = settings.getColumnNames().indexOf("petal length");
84      int petalWidthColumn = settings.getColumnNames().indexOf("petal width");
85      double mostImportant = Doubles.max(importances);
86      assertTrue(importances[petalLengthColumn] == mostImportant ||
87                     importances[petalWidthColumn] == mostImportant);
88  
89  
90      DecisionTree[] trees = forest.getTrees();
91      assertEquals(2, trees.length);
92  
93      Feature[] features = {
94          //5.9,3.0,5.1,1.8,Iris-virginica
95          NumericFeature.forValue(5.9f),
96          NumericFeature.forValue(3.0f),
97          NumericFeature.forValue(5.1f),
98          NumericFeature.forValue(1.8f),
99      };
100     Example example = new Example(null, features);
101     CategoricalPrediction prediction = (CategoricalPrediction) forest.classify(example);
102     int expectedCategory = targetCategoryValueMapping.get("Iris-virginica");
103     assertEquals(expectedCategory, prediction.getMostProbableCategoryID());
104     assertEquals(prediction.getCategoryProbabilities()[expectedCategory],
105                  Floats.max(prediction.getCategoryProbabilities()));
106   }
