16  package com.cloudera.oryx.kmeans.common;
17  
18  import java.util.ArrayList;
19  import java.util.Arrays;
20  import java.util.Collection;
21  import java.util.Collections;
22  import java.util.List;
23  
24  import org.apache.commons.math3.random.RandomGenerator;
25  import org.junit.Test;
26  
27  import com.google.common.base.Function;
28  import com.google.common.collect.Lists;
29  
30  import com.cloudera.oryx.common.OryxTest;
31  import com.cloudera.oryx.common.random.RandomManager;
32  
33  public final class WeightedTest extends OryxTest {
34  
35    private static final class ThingFn<T> implements Function<Weighted<T>, T> {
36      @Override
37      public T apply(Weighted<T> arg0) {
38        return arg0.thing();
39      }
40    }
41    
42    @Test
43    public void testBasic() throws Exception {
44      Collection<Weighted<Integer>> things = new ArrayList<>();
45      RandomGenerator rand = RandomManager.getRandom();
46      for (int i = 0; i < 50; i++) {
47        things.add(new Weighted<>(i, rand.nextDouble()));
48      }
49      List<Weighted<Integer>> s = Weighted.sample(things, 5, rand);
50      assertEquals(Arrays.asList(31, 46, 49, 23, 14), Lists.transform(s, new ThingFn<Integer>()));
51    }
52    
53    @Test
54    public void testEmpty() throws Exception {
55      RandomGenerator rand = RandomManager.getRandom();
56      assertEquals(0, Weighted.sample(Collections.<Weighted<Long>>emptyList(), 10, rand).size());
57    }
58    
59    @Test
60    public void testSmallerCollectionThanSize() throws Exception {
61      RandomGenerator rand = RandomManager.getRandom();
62      Collection<Weighted<Integer>> things = new ArrayList<>();
63      for (int i = 0; i < 5; i++) {
64        things.add(new Weighted<>(i, rand.nextDouble()));
65      }
66      List<Weighted<Integer>> s = Weighted.sample(things, 10, rand);
67      assertEquals(Arrays.asList(3, 4, 1, 0, 2), Lists.transform(s, new ThingFn<Integer>()));
68    }
69  }
