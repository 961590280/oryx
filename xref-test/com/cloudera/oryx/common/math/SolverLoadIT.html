32  /**
33   * Tests performance of {@link LinearSystemSolver#getSolver(RealMatrix)}.
34   * 
35   * @author Sean Owen
36   */
37  public final class SolverLoadIT extends OryxTest {
38    
39    private static final Logger log = LoggerFactory.getLogger(SolverLoadIT.class);
40  
41    @Ignore("Disable load testing by default")
42    @Test
43    public void testLoad() {
44      RealMatrix symmetric = randomSymmetricMatrix(500);
45      Stopwatch stopwatch = new Stopwatch().start();
46      int iterations = 100;
47      for (int i = 0; i < iterations; i++) {
48        MatrixUtils.getSolver(symmetric);
49      }
50      stopwatch.stop();
51      // long elapsedMS = stopwatch.elapsed(TimeUnit.MILLISECONDS);
52      long elapsedMS = stopwatch.elapsedTime(TimeUnit.MILLISECONDS);
53      log.info("{}ms elapsed for {} iterations", elapsedMS, iterations);
54      assertTrue(elapsedMS < 800 * iterations);
55    }
56    
57    private static RealMatrix randomSymmetricMatrix(int dimension) {
58      RandomGenerator random = RandomManager.getRandom();
59      RealMatrix symmetric = new Array2DRowRealMatrix(dimension, dimension);
60      for (int j = 0; j < dimension; j++) {
61        // Diagonal
62        symmetric.setEntry(j, j, random.nextDouble());
63        for (int k = j + 1; k < dimension; k++) {
64          // Off-diagonal
65          double d = random.nextDouble();
66          symmetric.setEntry(j, k, d);
67          symmetric.setEntry(k, j, d);          
68        }
69      }
70      return symmetric;
71    }
