71  /**
72   * <p>This is the runnable class which starts the Serving Layer and its Tomcat-based HTTP server. It is
73   * started with {@link #call()} and can be shut down with {@link #close()}. This implementation is used
74   * both in stand-alone local mode, and in a distributed mode cooperating with a Computation Layer.</p>
75   *
76   * <p>This program instantiates a Tomcat-based HTTP server exposing a REST-style API. It is available via
77   * HTTP, or HTTPS as well if "serving-layer.api.keystore-file" is set. It can also be password
78   * protected by setting "serving-layer.api.user-name" and
79   * "serving-layer.api.password".</p>
80   *
81   * <p>When run in local mode, the Serving Layer instance will compute a model locally and save it as files
82   * under a {@code model/} directory. It will be updated when the model is rebuilt.
83   * If the file is present at startup, it will be read to restore the server state, rather than re-reading
84   * CSV input in the directory and recomputing the model. Thus the file can be saved and restored as a
85   * way of preserving and recalling the server's state of learning.</p>
86   *
87   * <p>Example of running in local mode:</p>
88   *
89   * <p>{@code java -jar oryx-serving-x.y.jar}</p>
90   *
91   * <p>(with an example of JVM tuning flags:)</p>
92   *
93   * <p>{@code java -Xmx1g -XX:NewRatio=12 -XX:+UseParallelOldGC -jar oryx-serving-x.y.jar}</p>
94   *
95   * <p>Finally, some more advanced tuning parameters are available. These are system parameters, set with
96   * {@code -Dproperty=value}.</p>
97   *
98   * @author Sean Owen
99   */
100 public final class Runner implements Callable<Object>, Closeable {

132     this.noSuchBaseDir.deleteOnExit();

135   public static void main(String[] args) throws Exception {
136     try (Runner runner = new Runner()) {
137       SignalManager.register(new Runnable() {
138         @Override
139         public void run() {
140           runner.close();
141         }
142       }, SignalType.INT, SignalType.TERM);
143       runner.call();
144       runner.await();
145     }
146   }

148   @Override
149   public synchronized Object call() throws IOException {
150 
151     if (closed) {
152       return Boolean.FALSE;
153     }
154 
155     MemoryHandler.setSensibleLogFormat();
156     java.util.logging.Logger.getLogger("").addHandler(new MemoryHandler());
157 
158     System.setProperty("org.apache.tomcat.util.buf.UDecoder.ALLOW_ENCODED_SLASH", "true");
159 
160     Tomcat tomcat = new Tomcat();
161     Connector connector = makeConnector();
162     configureTomcat(tomcat, connector);
163     configureEngine(tomcat.getEngine());
164     configureServer(tomcat.getServer());
165     configureHost(tomcat.getHost());
166     Context context = makeContext(tomcat, noSuchBaseDir);
167 
168     AbstractOryxServingInitListener apiListener = chooseAPIListener();
169     context.addApplicationListener(apiListener.getClass().getName());
170 
171     AbstractOryxServingInitListener.addServlet(context, new style_jspx(), "/style.jspx");
172     AbstractOryxServingInitListener.addServlet(context, new index_jspx(), "/index.jspx");
173     AbstractOryxServingInitListener.addServlet(context, new status_jspx(), "/status.jspx");
174     AbstractOryxServingInitListener.addServlet(context, new error_jspx(), "/error.jspx");
175     AbstractOryxServingInitListener.addServlet(context, new LogServlet(), "/log.txt");
176     AbstractOryxServingInitListener.addServlet(context, new ConfigServlet(), "/config.json");
177     AbstractOryxServingInitListener.addServlet(context, apiListener.getApiJSPXFile(), "/test.jspx");
178     apiListener.addServlets(context);
179 
180     try {
181       tomcat.start();
182     } catch (LifecycleException le) {
183       throw new IOException(le);
184     }
185     this.tomcat = tomcat;
186     return Boolean.TRUE;
187   }
188 
189   /**
190    * Blocks and waits until the server shuts down.
191    */
192   void await() {
193     Server server = null;
194     synchronized (this) {
195       if (tomcat != null) {
196         server = tomcat.getServer();
197       }
198     }
199     if (server != null) {
200       server.await();
201     }
202   }
203 
204   @Override
205   public synchronized void close() {
206     if (!closed) {
207       closed = true;
208       if (tomcat != null) {
209         try {
210           tomcat.stop();
211           tomcat.destroy();
212         } catch (LifecycleException le) {
213           log.warn("Unexpected error while stopping", le);
214         }
215         try {
216           IOUtils.deleteRecursively(noSuchBaseDir);
217         } catch (IOException e) {
218           log.warn("Failed to delete {}", noSuchBaseDir);
219         }
220       }
221     }
222   }

224   private void configureTomcat(Tomcat tomcat, Connector connector) {
225     tomcat.setBaseDir(noSuchBaseDir.getAbsolutePath());
226     tomcat.setConnector(connector);
227     tomcat.getService().addConnector(connector);
228   }

230   private void configureEngine(Engine engine) {
231     APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
232     String userName = apiSettings.getUserName();
233     String password = apiSettings.getPassword();
234     if (userName != null && password != null) {
235       InMemoryRealm realm = new InMemoryRealm();
236       realm.addUser(userName, password);
237       engine.setRealm(realm);
238     }
239   }

241   private static void configureServer(Server server) {
242     LifecycleListener jasperListener = new JasperListener();
243     server.addLifecycleListener(jasperListener);
244     jasperListener.lifecycleEvent(new LifecycleEvent(server, Lifecycle.BEFORE_INIT_EVENT, null));
245     server.addLifecycleListener(new JreMemoryLeakPreventionListener());
246     server.addLifecycleListener(new ThreadLocalLeakPreventionListener());
247   }

249   private static void configureHost(Host host) {
250     host.setAutoDeploy(false);
251   }

253   private Connector makeConnector() {
254     Connector connector = new Connector("org.apache.coyote.http11.Http11NioProtocol");
255     APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
256 
257     File keystoreFile = apiSettings.getKeystoreFile();
258     String keystorePassword = apiSettings.getKeystorePassword();
259     if (keystoreFile == null && keystorePassword == null) {
260       // HTTP connector
261       connector.setPort(apiSettings.getPort());
262       connector.setSecure(false);
263       connector.setScheme("http");
264 
265     } else {
266 
267       // HTTPS connector
268       connector.setPort(apiSettings.getSecurePort());
269       connector.setSecure(true);
270       connector.setScheme("https");
271       connector.setAttribute("SSLEnabled", "true");
272       String protocol = chooseSSLProtocol("TLSv1.2", "TLSv1.1", "TLSv1");
273       if (protocol != null) {
274         connector.setAttribute("sslProtocol", protocol);
275       }
276       if (keystoreFile != null) {
277         connector.setAttribute("keystoreFile", keystoreFile.getAbsoluteFile());
278       }
279       connector.setAttribute("keystorePass", keystorePassword);
280     }
281 
282     // Keep quiet about the server type
283     connector.setXpoweredBy(false);
284     connector.setAttribute("server", "Oryx");
285 
286     // Basic tuning params:
287     connector.setAttribute("maxThreads", 400);
288     connector.setAttribute("acceptCount", 50);
289     //connector.setAttribute("connectionTimeout", 2000);
290     connector.setAttribute("maxKeepAliveRequests", 100);
291 
292     // Avoid running out of ephemeral ports under heavy load?
293     connector.setAttribute("socket.soReuseAddress", true);
294     
295     connector.setMaxPostSize(0);
296     connector.setAttribute("disableUploadTimeout", false);
297 
298     // Allow long URLs
299     connector.setAttribute("maxHttpHeaderSize", 65536);
300 
301     return connector;
302   }

304   private static String chooseSSLProtocol(String... protocols) {
305     for (String protocol : protocols) {
306       try {
307         SSLContext.getInstance(protocol);
308         return protocol;
309       } catch (NoSuchAlgorithmException ignored) {
310         // continue
311       }
312     }
313     return null;
314   }

316   private Context makeContext(Tomcat tomcat, File noSuchBaseDir) throws IOException {
317 
318     File contextPath = new File(noSuchBaseDir, "context");
319     IOUtils.mkdirs(contextPath);
320 
321     APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
322     String contextPathURIBase = config.getString("serving-layer.api.context-path");
323     if (contextPathURIBase == null || contextPathURIBase.isEmpty() || "/".equals(contextPathURIBase)) {
324       contextPathURIBase = "";
325     }
326     Context context = tomcat.addContext(contextPathURIBase, contextPath.getAbsolutePath());
327 
328     context.setWebappVersion("3.0");
329     context.addWelcomeFile("index.jspx");
330     addErrorPages(context);
331 
332     boolean needHTTPS = apiSettings.isSecure();
333     boolean needAuthentication = apiSettings.getUserName() != null;
334 
335     if (needHTTPS || needAuthentication) {
336 
337       SecurityCollection securityCollection = new SecurityCollection();
338       securityCollection.addPattern("/*");
339       SecurityConstraint securityConstraint = new SecurityConstraint();
340       securityConstraint.addCollection(securityCollection);
341 
342       if (needHTTPS) {
343         securityConstraint.setUserConstraint("CONFIDENTIAL");
344       }
345 
346       if (needAuthentication) {
347 
348         LoginConfig loginConfig = new LoginConfig();
349         loginConfig.setAuthMethod("DIGEST");
350         loginConfig.setRealmName(InMemoryRealm.NAME);
351         context.setLoginConfig(loginConfig);
352 
353         securityConstraint.addAuthRole(InMemoryRealm.AUTH_ROLE);
354 
355         context.addSecurityRole(InMemoryRealm.AUTH_ROLE);
356         DigestAuthenticator authenticator = new DigestAuthenticator();
357         authenticator.setNonceValidity(10 * 1000L); // Shorten from 5 minutes to 10 seconds
358         authenticator.setNonceCacheSize(20000); // Increase from 1000 to 20000
359         context.getPipeline().addValve(authenticator);
360       }
361 
362       context.addConstraint(securityConstraint);
363     }
364 
365     context.setCookies(false);
366 
367     return context;
368   }

370   private AbstractOryxServingInitListener chooseAPIListener() {
371     Class<? extends AbstractOryxServingInitListener> listenerClass = LISTENERS.get(config.getString("model.type"));
372     Preconditions.checkNotNull(listenerClass,
373                                "Unspecified or unsupported model type: %s",
374                                config.getString("model.type"));
375     return ClassUtils.loadInstanceOf(listenerClass);
376   }

378   private static void addErrorPages(Context context) {
379     for (int errorCode : ERROR_PAGE_STATUSES) {
380       ErrorPage errorPage = new ErrorPage();
381       errorPage.setErrorCode(errorCode);
382       errorPage.setLocation("/error.jspx");
383       context.addErrorPage(errorPage);
384     }
385     ErrorPage errorPage = new ErrorPage();
386     errorPage.setExceptionType(Throwable.class.getName());
387     errorPage.setLocation("/error.jspx");
388     context.addErrorPage(errorPage);
389   }
