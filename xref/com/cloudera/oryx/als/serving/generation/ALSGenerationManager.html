16  package com.cloudera.oryx.als.serving.generation;
17  
18  import java.io.File;
19  import java.io.IOException;
20  import java.io.Writer;
21  
22  import com.typesafe.config.Config;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import com.cloudera.oryx.als.common.StringLongMapping;
27  import com.cloudera.oryx.common.collection.LongSet;
28  import com.cloudera.oryx.common.math.SolverException;
29  import com.cloudera.oryx.common.io.DelimitedDataUtils;
30  import com.cloudera.oryx.common.settings.ConfigUtils;
31  import com.cloudera.oryx.serving.generation.GenerationManager;

33  /**
34   * An implementation of {@link ALSGenerationManager} is responsible for interacting with successive generations of the
35   * underlying recommender model. It sends updates to the component responsible for computing the model,
36   * and manages switching in new models when they become available.
37   * 
38   * @author Sean Owen
39   */
40  public final class ALSGenerationManager extends GenerationManager {
41  
42    private static final Logger log = LoggerFactory.getLogger(ALSGenerationManager.class);
43  
44    private int modelGeneration;
45    private Generation currentGeneration;
46    private final LongSet recentlyActiveUsers;
47    private final LongSet recentlyActiveItems;
48    private final GenerationLoader loader;
49    private final boolean disableWriteUpdates;
50  
51    public ALSGenerationManager(File appendTempDir) throws IOException {
52      super(appendTempDir);
53      modelGeneration = NO_GENERATION;
54      recentlyActiveUsers = new LongSet();
55      recentlyActiveItems = new LongSet();
56      Config config = ConfigUtils.getDefaultConfig();
57      loader = new GenerationLoader(
58          config.getString("model.instance-dir"), recentlyActiveUsers, recentlyActiveItems, this);
59      disableWriteUpdates = config.getBoolean("serving-layer.disable-write-updates");
60    }
61  
62    /**
63     * @return an instance of the latest {@link Generation} that has been made available by the
64     * implementation.
65     */
66    public Generation getCurrentGeneration() {
67      return currentGeneration;
68    }
69  
70    /**
71     * Sends a new user / item association to the component responsible for later recomputing
72     * the model based on this, and other, updates.
73     *
74     * @param userID user involved in new association
75     * @param itemID item involved
76     * @param value strength of the user/item association
77     * @throws IOException if an error occurs while sending the update
78     */
79    public void append(String userID, String itemID, float value) throws IOException {
80      if (!disableWriteUpdates) {
81        StringBuilder line = new StringBuilder(32);
82        line.append(DelimitedDataUtils.encode(',', userID, itemID, Float.toString(value))).append('\n');
83        doAppend(line, userID, itemID);
84      }
85    }
86  
87    /**
88     * Records that the user-item association should be removed. This is different from recording a
89     * negative association.
90     *
91     * @param userID user involved in new association
92     * @param itemID item involved
93     * @throws IOException if an error occurs while sending the update
94     */
95    public void remove(String userID, String itemID) throws IOException {
96      if (!disableWriteUpdates) {
97        StringBuilder line = new StringBuilder(32);
98        line.append(DelimitedDataUtils.encode(',', userID, itemID, "")).append('\n');
99        doAppend(line, userID, itemID);
100     }
101   }
102 
103   private synchronized void doAppend(CharSequence line, String userID, String itemID) throws IOException {
104     Writer appender = getAppender();
105     if (appender != null) {
106       appender.append(line);
107     }
108     // User ID reverse mapping is not important to serving layer
109     long numericUserID = StringLongMapping.toLong(userID);
110     long numericItemID;
111     if (currentGeneration == null) {
112       numericItemID = StringLongMapping.toLong(itemID);
113     } else {
114       numericItemID = currentGeneration.getIDMapping().add(itemID);
115     }
116     recentlyActiveUsers.add(numericUserID);
117     recentlyActiveItems.add(numericItemID);
118     decrementCountdownToUpload();
119   }
120 
121   @Override
122   protected void loadRecentModel(int mostRecentModelGeneration) throws IOException {
123     if (mostRecentModelGeneration <= modelGeneration) {
124       return;
125     }
126     if (modelGeneration == NO_GENERATION) {
127       log.info("Most recent generation {} is the first available one", mostRecentModelGeneration);
128     } else {
129       log.info("Most recent generation {} is newer than current {}",
130                mostRecentModelGeneration, modelGeneration);
131     }
132     try {
133 
134       Generation theCurrentGeneration = currentGeneration;
135       if (theCurrentGeneration == null) {
136         theCurrentGeneration = new Generation();
137       }
138 
139       loader.loadModel(mostRecentModelGeneration, theCurrentGeneration);
140 
141       modelGeneration = mostRecentModelGeneration;
142       currentGeneration = theCurrentGeneration;
143 
144     } catch (OutOfMemoryError oome) {
145       log.warn("Increase heap size with -Xmx, decrease new generation size with larger " +
146                    "-XX:NewRatio value, and/or use -XX:+UseCompressedOops");
147       currentGeneration = null;
148       throw oome;
149     } catch (SolverException ignored) {
150       log.warn("Unable to compute a valid generation yet; waiting for more data");
151       currentGeneration = null;
152     }
153   }
154 
155 }
