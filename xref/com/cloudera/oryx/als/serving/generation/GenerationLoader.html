49    private static final Logger log = LoggerFactory.getLogger(GenerationLoader.class);
50  
51    private final String instanceDir;
52    private final LongSet recentlyActiveUsers;
53    private final LongSet recentlyActiveItems;
54    private final Object lockForRecent;
55    
56    GenerationLoader(String instanceDir,
57                     LongSet recentlyActiveUsers,
58                     LongSet recentlyActiveItems,
59                     Object lockForRecent) {
60      this.instanceDir = instanceDir;
61      this.recentlyActiveUsers = recentlyActiveUsers;
62      this.recentlyActiveItems = recentlyActiveItems;
63      // This must be acquired to access the 'recent' fields above:
64      this.lockForRecent = lockForRecent;
65    }
66  
67    void loadModel(int generationID, Generation currentGeneration) throws IOException {
68  
69      File modelPMMLFile = File.createTempFile("oryx-model", ".pmml.gz");
70      modelPMMLFile.deleteOnExit();
71      IOUtils.delete(modelPMMLFile);
72  
73      String generationPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
74      String modelPMMLKey = generationPrefix + "model.pmml.gz";
75      Store.get().download(modelPMMLKey, modelPMMLFile);
76      log.info("Loading model description from {}", modelPMMLKey);
77  
78      ALSModelDescription modelDescription = ALSModelDescription.read(modelPMMLFile);
79      IOUtils.delete(modelPMMLFile);
80  
81      Collection<Future<Object>> futures = new ArrayList<>();
82      // Limit this fairly sharply to 2 so as to not saturate the network link
83      ExecutorService executor = ExecutorUtils.buildExecutor("LoadModel", 2);
84  
85      LongSet loadedUserIDs;
86      LongSet loadedItemIDs;
87      LongSet loadedUserIDsForKnownItems;
88      try {
89        loadedUserIDs = loadXOrY(generationPrefix, modelDescription, true, currentGeneration, futures, executor);
90        loadedItemIDs = loadXOrY(generationPrefix, modelDescription, false, currentGeneration, futures, executor);
91  
92        if (currentGeneration.getKnownItemIDs() == null) {
93          loadedUserIDsForKnownItems = null;
94        } else {
95          loadedUserIDsForKnownItems =
96              loadKnownItemIDs(generationPrefix,
97                               modelDescription,
98                               currentGeneration,
99                               futures,
100                              executor);
101       }
102 
103       loadIDMapping(generationPrefix, modelDescription, currentGeneration, futures, executor);
104 
105       ExecutorUtils.getResults(futures);
106       log.info("Finished all load tasks");
107       
108     } finally {
109       ExecutorUtils.shutdownNowAndAwait(executor);
110     }
111 
112     log.info("Pruning old entries...");
113     synchronized (lockForRecent) {
114       removeNotUpdated(currentGeneration.getX().keySetIterator(),
115                        loadedUserIDs,
116                        recentlyActiveUsers,
117                        currentGeneration.getXLock().writeLock());
118       removeNotUpdated(currentGeneration.getY().keySetIterator(),
119                        loadedItemIDs,
120                        recentlyActiveItems,
121                        currentGeneration.getYLock().writeLock());
122       if (loadedUserIDsForKnownItems != null && currentGeneration.getKnownItemIDs() != null) {
123         removeNotUpdated(currentGeneration.getKnownItemIDs().keySetIterator(),
124                          loadedUserIDsForKnownItems,
125                          recentlyActiveUsers,
126                          currentGeneration.getKnownItemLock().writeLock());
127       }
128       this.recentlyActiveItems.clear();
129       this.recentlyActiveUsers.clear();
130     }
131 
132     log.info("Recomputing generation state...");
133     currentGeneration.recomputeState();
134 
135     log.info("All model elements loaded, {} users and {} items", 
136              currentGeneration.getNumUsers(), currentGeneration.getNumItems());
137   }

139   private static LongSet loadXOrY(String generationPrefix,
140                                     ALSModelDescription modelDescription,
141                                     boolean isX,
142                                     Generation generation,
143                                     Collection<Future<Object>> futures,
144                                     ExecutorService executor) throws IOException {
145 
146     String xOrYPrefix = generationPrefix + (isX ? modelDescription.getXPath() : modelDescription.getYPath());
147     final LongSet loadedIDs = new LongSet();
148     
149     final Lock writeLock = isX ? generation.getXLock().writeLock() : generation.getYLock().writeLock();
150     final LongObjectMap<float[]> xOrYMatrix = isX ? generation.getX() : generation.getY();
151 
152     for (final String xOrYFilePrefix : Store.get().list(xOrYPrefix, true)) {
153       futures.add(executor.submit(new Callable<Object>() {
154         @Override
155         public Void call() throws IOException {
156           for (String line : new FileLineIterable(Store.get().readFrom(xOrYFilePrefix))) {
157 
158             int tab = line.indexOf('\t');
159             Preconditions.checkArgument(tab >= 0, "Bad input line in %s: %s", xOrYFilePrefix, line);
160             long id = Long.parseLong(line.substring(0, tab));
161 
162             float[] elements = DataUtils.readFeatureVector(line.substring(tab + 1));
163 
164             writeLock.lock();
165             try {
166               xOrYMatrix.put(id, elements);
167               loadedIDs.add(id);
168             } finally {
169               writeLock.unlock();
170             }
171           }
172           log.info("Loaded feature vectors from {}", xOrYFilePrefix);
173           return null;
174         }
175       }));
176     }
177 
178     return loadedIDs;
179   }

182   private static LongSet loadKnownItemIDs(String generationPrefix,
183                                             ALSModelDescription modelDescription,
184                                             final Generation generation,
185                                             Collection<Future<Object>> futures,
186                                             ExecutorService executor) throws IOException {
187     final LongSet loadedIDs = new LongSet();
188     String knownItemsPrefix = generationPrefix + modelDescription.getKnownItemsPath();
189     for (final String knownItemFilePrefix : Store.get().list(knownItemsPrefix, true)) {
190       futures.add(executor.submit(new Callable<Object>() {
191         @Override
192         public Void call() throws IOException {
193           for (String line : new FileLineIterable(Store.get().readFrom(knownItemFilePrefix))) {
194             int tab = line.indexOf('\t');
195             Preconditions.checkArgument(tab >= 0, "Bad input line in %s: %s", knownItemFilePrefix, line);
196             long userID = Long.parseLong(line.substring(0, tab));
197             LongSet itemIDs = stringToSet(line.substring(tab + 1));
198             Lock writeLock = generation.getKnownItemLock().writeLock();
199             LongObjectMap<LongSet> knownItems = generation.getKnownItemIDs();
200             writeLock.lock();
201             try {
202               knownItems.put(userID, itemIDs);
203               loadedIDs.add(userID);
204             } finally {
205               writeLock.unlock();
206             }
207           }
208           log.info("Loaded known items from {}", knownItemFilePrefix);
209           return null;
210         }
211       }));
212     }
213     return loadedIDs;
214   }

216   private static LongSet stringToSet(CharSequence values) {
217     LongSet result = new LongSet();
218     for (String valueString : DelimitedDataUtils.decode(values, ',')) {
219       result.add(Long.parseLong(valueString));
220     }
221     return result;
222   }
223 
224   private static void removeNotUpdated(LongPrimitiveIterator it,
225                                        LongSet updated,
226                                        LongSet recentlyActive,
227                                        Lock writeLock) {
228     writeLock.lock();
229     try {
230       while (it.hasNext()) {
231         long id = it.nextLong();
232         if (!updated.contains(id) && !recentlyActive.contains(id)) {
233           it.remove();
234         }
235       }
236     } finally {
237       writeLock.unlock();
238     }
239   }

241   private static void loadIDMapping(String generationPrefix,
242                                     ALSModelDescription modelDescription,
243                                     final Generation generation,
244                                     Collection<Future<Object>> futures,
245                                     ExecutorService executor) throws IOException {
246     String idMappingPrefix = generationPrefix + modelDescription.getIDMappingPath();
247     for (final String prefix : Store.get().list(idMappingPrefix, true)) {
248       futures.add(executor.submit(new Callable<Object>() {
249         @Override
250         public Void call() throws IOException {
251           for (CharSequence line : new FileLineIterable(Store.get().readFrom(prefix))) {
252             String[] columns = DelimitedDataUtils.decode(line, ',');
253             long numericID = Long.parseLong(columns[0]);
254             String id = columns[1];
255             Lock writeLock = generation.getKnownItemLock().writeLock();
256             StringLongMapping idMapping = generation.getIDMapping();
257             writeLock.lock();
258             try {
259               idMapping.addMapping(id, numericID);
260             } finally {
261               writeLock.unlock();
262             }
263           }
264           return null;
265         }
266       }));
267     }
268   }
