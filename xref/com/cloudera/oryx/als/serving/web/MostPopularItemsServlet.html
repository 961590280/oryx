16  package com.cloudera.oryx.als.serving.web;
17  
18  import java.io.IOException;
19  import javax.servlet.http.HttpServletRequest;
20  import javax.servlet.http.HttpServletResponse;
21  
22  import com.cloudera.oryx.als.common.rescorer.Rescorer;
23  import com.cloudera.oryx.als.common.NotReadyException;
24  import com.cloudera.oryx.als.common.OryxRecommender;
25  import com.cloudera.oryx.als.common.rescorer.RescorerProvider;
26  
27  /**
28   * <p>Responds to a GET request to {@code /mostPopularItems(?howMany=n)(&offset=o)}
29   * and in turn calls {@link OryxRecommender#mostPopularItems(int)}.
30   * {@code offset} is an offset into the entire list of results; {@code howMany} is the desired
31   * number of results to return from there. For example, {@code offset=30} and {@code howMany=5}
32   * will cause the implementation to retrieve 35 results internally and output the last 5.
33   * If {@code howMany} is not specified, defaults to {@link AbstractALSServlet#DEFAULT_HOW_MANY}.
34   * {@code offset} defaults to 0.</p>
35   *
36   * <p>Output is as in {@link RecommendServlet}.</p>
37   *
38   * @author Sean Owen
39   */
40  public final class MostPopularItemsServlet extends AbstractALSServlet {
41  
42    @Override
43    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
44      OryxRecommender recommender = getRecommender();
45      RescorerProvider rescorerProvider = getRescorerProvider();
46      try {
47        Rescorer rescorer = rescorerProvider == null ? null :
48            rescorerProvider.getMostPopularItemsRescorer(recommender, getRescorerParams(request));
49        outputALSResult(request, response, recommender.mostPopularItems(getNumResultsToFetch(request), rescorer));
50      } catch (NotReadyException nre) {
51        response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE, nre.toString());
52      } catch (IllegalArgumentException iae) {
53        response.sendError(HttpServletResponse.SC_BAD_REQUEST, iae.toString());
54      } catch (UnsupportedOperationException uoe) {
55        response.sendError(HttpServletResponse.SC_BAD_REQUEST, uoe.toString());
56      }
57    }
58  
59  }
