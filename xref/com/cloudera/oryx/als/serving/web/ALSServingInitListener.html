38  /**
39   * <p>This servlet lifecycle listener makes sure that the shared
40   * {@link com.cloudera.oryx.als.common.OryxRecommender} instance
41   * is initialized at startup, along with related objects, and shut down when the container
42   * is destroyed.</p>
43   *
44   * @author Sean Owen
45   */
46  public final class ALSServingInitListener extends AbstractOryxServingInitListener {
47  
48    private static final Logger log = LoggerFactory.getLogger(ALSServingInitListener.class);
49  
50    @Override
51    public void contextInitialized(ServletContextEvent event) {
52      super.contextInitialized(event);
53      ServletContext context = event.getServletContext();
54  
55      File localInputDir = getLocalInputDir();
56      ServerRecommender recommender;
57      try {
58        recommender = new ServerRecommender(localInputDir);
59      } catch (IOException ioe) {
60        throw new IllegalStateException(ioe);
61      }
62      context.setAttribute(AbstractALSServlet.RECOMMENDER_KEY, recommender);
63  
64      Config config = ConfigUtils.getDefaultConfig();
65      if (config.hasPath("serving-layer.rescorer-provider-class")) {
66        String rescorerProviderClassNames = config.getString("serving-layer.rescorer-provider-class");
67        RescorerProvider rescorerProvider =
68            AbstractRescorerProvider.loadRescorerProviders(rescorerProviderClassNames);
69        if (rescorerProvider != null) {
70          context.setAttribute(AbstractALSServlet.RESCORER_PROVIDER_KEY, rescorerProvider);
71        }
72      }
73    }
74  
75    @Override
76    public void contextDestroyed(ServletContextEvent event) {
77      ServletContext context = event.getServletContext();
78      Closeable recommender = (Closeable) context.getAttribute(AbstractALSServlet.RECOMMENDER_KEY);
79      if (recommender != null) {
80        try {
81          recommender.close();
82        } catch (IOException e) {
83          log.warn("Unexpected error while closing", e);
84        }
85      }
86    }
87  
88    @Override
89    public HttpServlet getApiJSPXFile() {
90      return new test_002dals_jspx();
91    }
92  
93    @Override
94    public void addServlets(Context context) {
95      addServlet(context, new RecommendServlet(), "/recommend/*");
96      addServlet(context, new RecommendToManyServlet(), "/recommendToMany/*");
97      addServlet(context, new RecommendToAnonymousServlet(), "/recommendToAnonymous/*");
98      addServlet(context, new SimilarityServlet(), "/similarity/*");
99      addServlet(context, new SimilarityToItemServlet(), "/similarityToItem/*");
100     addServlet(context, new EstimateServlet(), "/estimate/*");
101     addServlet(context, new EstimateForAnonymousServlet(), "/estimateForAnonymous/*");
102     addServlet(context, new BecauseServlet(), "/because/*");
103     addServlet(context, new ReadyServlet(), "/ready/*");
104     addServlet(context, new MostPopularItemsServlet(), "/mostPopularItems/*");
105     addServlet(context, new MostActiveUsersServlet(), "/mostActiveUsers");
106     addServlet(context, new PopularRepresentativeItemsServlet(), "/popularRepresentativeItems/*");
107     addServlet(context, new AllItemIDsServlet(), "/item/allIDs");
108     addServlet(context, new AllUserIDsServlet(), "/user/allIDs");
109     addServlet(context, new KnownItemsServlet(), "/knownItems/*");
110     addServlet(context, new MostSurprisingServlet(), "/mostSurprising/*");
111     if (!ConfigUtils.getDefaultConfig().getBoolean("serving-layer.api.read-only")) {
112       addServlet(context, new PreferenceServlet(), "/pref/*");
113       Wrapper ingestWrapper = addServlet(context, new IngestServlet(), "/ingest/*");
114       ingestWrapper.setMultipartConfigElement(new MultipartConfigElement("/tmp"));
115       addServlet(context, new RefreshServlet(), "/refresh/*");
116     }
117   }
