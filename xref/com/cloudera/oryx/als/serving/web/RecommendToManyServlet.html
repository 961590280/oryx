32  /**
33   * <p>Responds to a GET request to
34   * {@code
35   * /recommendToMany/[userID1](/...)(?howMany=n)(&offset=o)(&considerKnownItems=true|false)(&rescorerParams=...)
36   * }
37   * and in turn calls
38   * {@link OryxRecommender#recommendToMany(String[], int, boolean, Rescorer)}.
39   * If {@code considerKnownItems} is not specified, it is considered {@code false}.
40   * {@code offset} is an offset into the entire list of results; {@code howMany} is the desired
41   * number of results to return from there. For example, {@code offset=30} and {@code howMany=5}
42   * will cause the implementation to retrieve 35 results internally and output the last 5.
43   * If {@code howMany} is not specified, defaults to {@link AbstractALSServlet#DEFAULT_HOW_MANY}.
44   * {@code offset} defaults to 0.</p>
45   *
46   * <p>Unknown user IDs are ignored, unless all are unknown, in which case a
47   * {@link HttpServletResponse#SC_BAD_REQUEST} status is returned.</p>
48   *
49   * <p>CSV output contains one recommendation per line, and each line is of the form {@code itemID, strength},
50   * like {@code 325, 0.53}. Strength is an opaque indicator of the relative quality of the recommendation.</p>
51   *
52   * @author Sean Owen
53   */
54  public final class RecommendToManyServlet extends AbstractALSServlet {
55    
56    @Override
57    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
58  
59      CharSequence pathInfo = request.getPathInfo();
60      if (pathInfo == null) {
61        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "No path");
62        return;
63      }
64      Iterator<String> pathComponents = SLASH.split(pathInfo).iterator();
65      Set<String> userIDSet = new HashSet<>();
66      try {
67        while (pathComponents.hasNext()) {
68          userIDSet.add(pathComponents.next());
69        }
70      } catch (NoSuchElementException nsee) {
71        response.sendError(HttpServletResponse.SC_BAD_REQUEST, nsee.toString());
72        return;
73      }
74  
75      String[] userIDs = userIDSet.toArray(new String[userIDSet.size()]);
76      unescapeSlashHack(userIDs);
77  
78      OryxRecommender recommender = getRecommender();
79      RescorerProvider rescorerProvider = getRescorerProvider();
80      try {
81        Rescorer rescorer = rescorerProvider == null ? null :
82            rescorerProvider.getRecommendRescorer(userIDs, recommender, getRescorerParams(request));
83        outputALSResult(request, response, recommender.recommendToMany(userIDs,
84                                                                       getNumResultsToFetch(request),
85                                                                       getConsiderKnownItems(request),
86                                                                       rescorer));
87      } catch (NoSuchUserException nsue) {
88        response.sendError(HttpServletResponse.SC_NOT_FOUND, nsue.toString());
89      } catch (NotReadyException nre) {
90        response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE, nre.toString());
91      } catch (IllegalArgumentException iae) {
92        response.sendError(HttpServletResponse.SC_BAD_REQUEST, iae.toString());
93      }
94    }
