48  /**
49   * <p>Implements the Alternating Least Squares algorithm described in
50   * <a href="http://www2.research.att.com/~yifanhu/PUB/cf.pdf">
51   * "Collaborative Filtering for Implicit Feedback Datasets"</a>
52   * by Yifan Hu, Yehuda Koren, and Chris Volinsky.</p>
53   *
54   * <p>This implementation varies in some small details; it does not use the same mechanism for explaining ratings
55   * for example and seeds the initial Y differently.</p>
56   *
57   * <p>Note that in this implementation, matrices are sparse and are implemented with a
58   * {@link com.cloudera.oryx.common.collection.LongObjectMap} of
59   * {@link com.cloudera.oryx.common.collection.LongFloatMap} so as to be able to use
60   * {@code long} keys. In many cases, a tall, skinny matrix is
61   * needed (sparse rows, dense columns). This is represented with
62   * {@link com.cloudera.oryx.common.collection.LongObjectMap} of {@code float[]}.</p>
63   *
64   * <p>This implementation implements essentially this function, expressed in Octave/Matlab:</p>
65   *
66   * {@code
67   * function [X, Y] = als(R, f, iterations)
68   *
69   *   alpha = 1;
70   *   lambda = 0.1 * alpha;
71   *   [m,n] = size(R);
72   *
73   *   P = R > 0;
74   *   Pnonzero = R != 0;
75   *   C = 1 + alpha * abs(R);
76   *
77   *   X = zeros(m,f);
78   *   Y = abs(stdnormal_rnd(n,f));
79   *   for i = 1:n
80   *     Y(i,:) /= norm(Y(i,:));
81   *   endfor
82   *
83   *   If = eye(f);
84   *   Im = eye(m);
85   *   In = eye(n);
86   *
87   *   for iter = 1:iterations
88   *
89   *     YTY = Y' * Y;
90   *     for u = 1:m
91   *       Cu = diag(C(u,:));
92   *       count = sum(Pnonzero(u,:));
93   *       YTCupu = Y' * Cu * P(u,:)';
94   *       YTCuY = YTY + Y' * (Cu - In) * Y;
95   *       Wu = pinv(YTCuY + lambda * count * If);
96   *       X(u,:) = (Wu * YTCupu)';
97   *     endfor
98   *
99   *     XTX = X' * X;
100  *     for i = 1:n
101  *       Ci = diag(C(:,i));
102  *       count = sum(Pnonzero(:,i));
103  *       XTCipi = X' * Ci * P(:,i);
104  *       XTCiX = XTX + X' * (Ci - Im) * X;
105  *       Wi = pinv(XTCiX + lambda * count * If);
106  *       Y(i,:) = (Wi * XTCipi)';
107  *     endfor
108  *
109  *   endfor
110  *
111  * end}
112  *
113  * @author Sean Owen
114  */
115 public final class AlternatingLeastSquares implements MatrixFactorizer {

119   private static final int WORK_UNIT_SIZE = 100;

123   private static final int MAX_FAR_FROM_VECTORS = 100000;

125   private final LongObjectMap<LongFloatMap> RbyRow;
126   private final LongObjectMap<LongFloatMap> RbyColumn;
127   private final int features;
128   private final double estimateErrorConvergenceThreshold;

134   /**
135    * @param RbyRow the input R matrix, indexed by row
136    * @param RbyColumn the input R matrix, indexed by column
137    * @param features number of features, must be positive
138    * @param estimateErrorConvergenceThreshold when the average absolute difference in estimated user-item
139    *   scores falls below this threshold between iterations, iterations will stop
140    * @param maxIterations caps the number of iterations run. If non-positive, there is no cap.
141    */
142   public AlternatingLeastSquares(LongObjectMap<LongFloatMap> RbyRow,
143                                  LongObjectMap<LongFloatMap> RbyColumn,
144                                  int features,
145                                  double estimateErrorConvergenceThreshold,
146                                  int maxIterations) {
147     Preconditions.checkNotNull(RbyRow);
148     Preconditions.checkNotNull(RbyColumn);
149     Preconditions.checkArgument(features > 0, "features must be positive: %s", features);
150     Preconditions.checkArgument(estimateErrorConvergenceThreshold > 0.0 && estimateErrorConvergenceThreshold < 1.0,
151                                 "threshold must be in (0,1): %s", estimateErrorConvergenceThreshold);
152     this.RbyRow = RbyRow;
153     this.RbyColumn = RbyColumn;
154     this.features = features;
155     this.estimateErrorConvergenceThreshold = estimateErrorConvergenceThreshold;
156     this.maxIterations = maxIterations;
157   }

170    * Sets the initial state of Y used in the computation, typically the Y from a previous
171    * computation. Call before {@link #call()}.

173   @Override
174   public void setPreviousY(LongObjectMap<float[]> previousY) {
175     this.previousY = previousY;

178   @Override
179   public Void call() throws ExecutionException, InterruptedException {
180 
181     X = new LongObjectMap<>(RbyRow.size());
182 
183     boolean randomY = previousY == null || previousY.isEmpty();
184     Y = constructInitialY(previousY);
185 
186     // This will be used to compute rows/columns in parallel during iteration
187     ExecutorService executor = ExecutorUtils.buildExecutor("ALS");
188 
189     RandomGenerator random = RandomManager.getRandom();
190     long[] testUserIDs = RandomUtils.chooseAboutNFromStream(NUM_USER_ITEMS_TO_TEST_CONVERGENCE, 
191                                                             RbyRow.keySetIterator(), 
192                                                             RbyRow.size(), 
193                                                             random);
194     long[] testItemIDs = RandomUtils.chooseAboutNFromStream(NUM_USER_ITEMS_TO_TEST_CONVERGENCE, 
195                                                             RbyColumn.keySetIterator(), 
196                                                             RbyColumn.size(), 
197                                                             random);
198     double[][] estimates = new double[testUserIDs.length][testItemIDs.length];
199     if (!X.isEmpty()) {
200       for (int i = 0; i < testUserIDs.length; i++) {
201         for (int j = 0; j < testItemIDs.length; j++) {
202           estimates[i][j] = SimpleVectorMath.dot(X.get(testUserIDs[i]), Y.get(testItemIDs[j])); 
203         }
204       }
205     }
206     // Otherwise X is empty because it's the first ever iteration. Estimates can be left at initial 0 value
207 
208     try {
209       int iterationNumber = 0;
210       while (true) {
211         iterateXFromY(executor);
212         iterateYFromX(executor);
213         DoubleWeightedMean averageAbsoluteEstimateDiff = new DoubleWeightedMean();
214         for (int i = 0; i < testUserIDs.length; i++) {
215           for (int j = 0; j < testItemIDs.length; j++) {
216             double newValue = SimpleVectorMath.dot(X.get(testUserIDs[i]), Y.get(testItemIDs[j]));            
217             double oldValue = estimates[i][j];
218             estimates[i][j] = newValue;
219             // Weight, simplistically, by newValue to emphasize effect of good recommendations.
220             // But that only makes sense where newValue > 0
221             if (newValue > 0.0f) {
222               averageAbsoluteEstimateDiff.increment(Math.abs(newValue - oldValue), newValue);
223             }
224           }
225         }
226 
227         iterationNumber++;
228         log.info("Finished iteration {}", iterationNumber);
229         if (maxIterations > 0 && iterationNumber >= maxIterations) {
230           log.info("Reached iteration limit");
231           break;
232         }
233 
234         long numSamples = averageAbsoluteEstimateDiff.getN();
235         double convergenceValue = averageAbsoluteEstimateDiff.getResult();
236         log.info("Avg absolute difference in estimate vs prior iteration over {} samples: {}",
237                  numSamples, convergenceValue);
238         if (Double.isNaN(convergenceValue) || Double.isInfinite(convergenceValue)) {
239           log.warn("Invalid convergence value, aborting iteration! {}", convergenceValue);
240           break;
241         }
242         // Don't converge after 1 iteration if starting from a random point
243         if (!(randomY && iterationNumber == 1) && convergenceValue < estimateErrorConvergenceThreshold) {
244           log.info("Converged");
245           break;
246         }
247       }
248     } finally {
249       ExecutorUtils.shutdownNowAndAwait(executor);
250     }
251     return null;
252   }

254   private LongObjectMap<float[]> constructInitialY(LongObjectMap<float[]> previousY) {
255 
256     RandomGenerator random = RandomManager.getRandom();
257     
258     LongObjectMap<float[]> randomY;
259     if (previousY == null || previousY.isEmpty()) {
260       // Common case: have to start from scratch
261       log.info("Starting from new, random Y matrix");      
262       randomY = new LongObjectMap<>(RbyColumn.size());
263       
264     } else {
265       
266       int oldFeatureCount = previousY.entrySet().iterator().next().getValue().length;
267       if (oldFeatureCount > features) {
268         // Fewer features, use some dimensions from prior larger number of features as-is
269         log.info("Feature count decreased to {}, projecting down previous Y matrix", features);
270         randomY = new LongObjectMap<>(previousY.size());
271         for (LongObjectMap.MapEntry<float[]> entry : previousY.entrySet()) {
272           float[] oldLargerVector = entry.getValue();
273           float[] newSmallerVector = new float[features];
274           System.arraycopy(oldLargerVector, 0, newSmallerVector, 0, newSmallerVector.length);
275           SimpleVectorMath.normalize(newSmallerVector);
276           randomY.put(entry.getKey(), newSmallerVector);
277         }
278         
279       } else if (oldFeatureCount < features) {
280         log.info("Feature count increased to {}, using previous Y matrix as subspace", features);
281         randomY = new LongObjectMap<>(previousY.size());
282         for (LongObjectMap.MapEntry<float[]> entry : previousY.entrySet()) {
283           float[] oldSmallerVector = entry.getValue();
284           float[] newLargerVector = new float[features];
285           System.arraycopy(oldSmallerVector, 0, newLargerVector, 0, oldSmallerVector.length);
286           // Fill in new dimensions with random values
287           for (int i = oldSmallerVector.length; i < newLargerVector.length; i++) {
288             newLargerVector[i] = Math.abs((float) random.nextGaussian());
289           }
290           SimpleVectorMath.normalize(newLargerVector);
291           randomY.put(entry.getKey(), newLargerVector);
292         }
293         
294       } else {
295         // Common case: previous generation is same number of features
296         log.info("Starting from previous generation's Y matrix");        
297         randomY = previousY;
298       }
299     }
300     
301     List<float[]> recentVectors = new ArrayList<>();
302     for (LongObjectMap.MapEntry<float[]> entry : randomY.entrySet()) {
303       if (recentVectors.size() >= MAX_FAR_FROM_VECTORS) {
304         break;
305       }
306       recentVectors.add(entry.getValue());
307     }
308     LongPrimitiveIterator it = RbyColumn.keySetIterator();
309     long count = 0;
310     while (it.hasNext()) {
311       long id = it.nextLong();
312       if (!randomY.containsKey(id)) {
313         float[] vector = RandomUtils.randomUnitVectorFarFrom(features, recentVectors, random);
314         randomY.put(id, vector);
315         if (recentVectors.size() < MAX_FAR_FROM_VECTORS) {
316           recentVectors.add(vector);
317         }
318       }
319       if (++count % LOG_INTERVAL == 0) {
320         log.info("Computed {} initial Y rows", count);
321       }
322     }
323     log.info("Constructed initial Y");
324     return randomY;
325   }

327   /**
328    * Runs one iteration to compute X from Y.
329    */
330   private void iterateXFromY(ExecutorService executor) throws ExecutionException, InterruptedException {
331 
332     RealMatrix YTY = MatrixUtils.transposeTimesSelf(Y);
333     Collection<Future<?>> futures = new ArrayList<>();
334     addWorkers(RbyRow, Y, YTY, X, executor, futures);
335 
336     int count = 0;
337     long total = 0;
338     for (Future<?> f : futures) {
339       f.get();
340       count += WORK_UNIT_SIZE;
341       if (count >= LOG_INTERVAL) {
342         total += count;
343         JVMEnvironment env = new JVMEnvironment();
344         log.info("{} X/tag rows computed ({}MB heap used of {}MB)",
345                  total, env.getUsedMemoryMB(), env.getMaxMemoryMB());
346         if (env.getPercentUsedMemory() > 95) {
347           log.warn("Memory is low. Increase heap size with -Xmx, decrease new generation size with larger " +
348                    "-XX:NewRatio value, and/or use -XX:+UseCompressedOops");
349         }
350         count = 0;
351       }
352     }
353   }

355   /**
356    * Runs one iteration to compute Y from X.
357    */
358   private void iterateYFromX(ExecutorService executor) throws ExecutionException, InterruptedException {
359 
360     RealMatrix XTX = MatrixUtils.transposeTimesSelf(X);
361     Collection<Future<?>> futures = new ArrayList<>();
362     addWorkers(RbyColumn, X, XTX, Y, executor, futures);
363 
364     int count = 0;
365     long total = 0;
366     for (Future<?> f : futures) {
367       f.get();
368       count += WORK_UNIT_SIZE;
369       if (count >= LOG_INTERVAL) {
370         total += count;
371         JVMEnvironment env = new JVMEnvironment();
372         log.info("{} Y/tag rows computed ({}MB heap used of {}MB)",
373                  total, env.getUsedMemoryMB(), env.getMaxMemoryMB());
374         if (env.getPercentUsedMemory() > 95) {
375           log.warn("Memory is low. Increase heap size with -Xmx, decrease new generation size with larger " +
376                    "-XX:NewRatio value, and/or use -XX:+UseCompressedOops");
377         }
378         count = 0;
379       }
380     }
381   }

383   private void addWorkers(LongObjectMap<LongFloatMap> R,
384                           LongObjectMap<float[]> M,
385                           RealMatrix MTM, 
386                           LongObjectMap<float[]> MTags,
387                           ExecutorService executor,                          
388                           Collection<Future<?>> futures) {
389     if (R != null) {
390       List<Pair<Long, LongFloatMap>> workUnit = new ArrayList<>(WORK_UNIT_SIZE);
391       for (LongObjectMap.MapEntry<LongFloatMap> entry : R.entrySet()) {
392         workUnit.add(new Pair<>(entry.getKey(), entry.getValue()));
393         if (workUnit.size() == WORK_UNIT_SIZE) {
394           futures.add(executor.submit(new Worker(features, M, MTM, MTags, workUnit)));
395           workUnit = new ArrayList<>(WORK_UNIT_SIZE);
396         }
397       }
398       if (!workUnit.isEmpty()) {
399         futures.add(executor.submit(new Worker(features, M, MTM, MTags, workUnit)));
400       }
401     }
402   }

404   private static final class Worker implements Callable<Object> {
405 
406     private final int features;
407     private final LongObjectMap<float[]> Y;
408     private final RealMatrix YTY;
409     private final LongObjectMap<float[]> X;
410     private final Iterable<Pair<Long, LongFloatMap>> workUnit;
411 
412     private Worker(int features,
413                    LongObjectMap<float[]> Y,
414                    RealMatrix YTY,
415                    LongObjectMap<float[]> X,
416                    Iterable<Pair<Long, LongFloatMap>> workUnit) {
417       this.features = features;
418       this.Y = Y;
419       this.YTY = YTY;
420       this.X = X;
421       this.workUnit = workUnit;
422     }
423 
424     @Override
425     public Void call() {
426 
427       Config config = ConfigUtils.getDefaultConfig();
428       double alpha = config.getDouble("model.alpha");
429       double lambda = config.getDouble("model.lambda") * alpha;
430       // This will cause the ALS algorithm to reconstruction the input matrix R, rather than the
431       // matrix P = R > 0 . Don't use this unless you understand it!
432       boolean reconstructRMatrix = config.getBoolean("model.reconstruct-r-matrix");
433       // Causes the loss function to exclude entries for any input pairs that do not appear in the
434       // input and are implicitly 0
435       // Likewise, don't touch this for now unless you know what it does.
436       boolean lossIgnoresUnspecified = config.getBoolean("model.loss-ignores-unspecified");
437 
438       int features = this.features;
439       // Each worker has a batch of rows to compute:
440       for (Pair<Long,LongFloatMap> work : workUnit) {
441 
442         // Row (column) in original R matrix containing total association value. For simplicity we will
443         // talk about users and rows only in the comments and variables. It's symmetric for columns / items.
444         // This is Ru:
445         LongFloatMap ru = work.getSecond();
446 
447         // Start computing Wu = (YT*Cu*Y + lambda*I) = (YT*Y + YT*(Cu-I)*Y + lambda*I),
448         // by first starting with a copy of YT * Y. Or, a variant on YT * Y, if LOSS_IGNORES_UNSPECIFIED is set
449         RealMatrix Wu =
450             lossIgnoresUnspecified ?
451             partialTransposeTimesSelf(Y, YTY.getRowDimension(), ru.keySetIterator()) : 
452             YTY.copy();
453 
454         double[][] WuData = MatrixUtils.accessMatrixDataDirectly(Wu);
455         double[] YTCupu = new double[features];
456 
457         for (LongFloatMap.MapEntry entry : ru.entrySet()) {
458 
459           double xu = entry.getValue();
460 
461           float[] vector = Y.get(entry.getKey());
462           if (vector == null) {
463             log.warn("No vector for {}. This should not happen. Continuing...", entry.getKey());
464             continue;
465           }
466 
467           // Wu and YTCupu
468           if (reconstructRMatrix) {
469             for (int row = 0; row < features; row++) {
470               YTCupu[row] += xu * vector[row];
471             }
472           } else {
473             double cu = 1.0 + alpha * Math.abs(xu);
474             for (int row = 0; row < features; row++) {
475               float vectorAtRow = vector[row];
476               double rowValue = vectorAtRow * (cu - 1.0);
477               double[] WuDataRow = WuData[row];              
478               for (int col = 0; col < features; col++) {
479                 WuDataRow[col] += rowValue * vector[col];
480                 //Wu.addToEntry(row, col, rowValue * vector[col]);
481               }
482               if (xu > 0.0) {
483                 YTCupu[row] += vectorAtRow * cu;
484               }
485             }
486           }
487 
488         }
489 
490         double lambdaTimesCount = lambda * ru.size();
491         for (int x = 0; x < features; x++) {
492           WuData[x][x] += lambdaTimesCount;          
493           //Wu.addToEntry(x, x, lambdaTimesCount);
494         }
495 
496         float[] xu = MatrixUtils.getSolver(Wu).solveDToF(YTCupu);
497 
498         // Store result:
499         synchronized (X) {
500           X.put(work.getFirst(), xu);
501         }
502 
503         // Process is identical for computing Y from X. Swap X in for Y, Y for X, i for u, etc.
504       }
505       return null;
506     }
507 
508     /**
509      * Like
510      * {@link MatrixUtils#transposeTimesSelf(com.cloudera.oryx.common.collection.LongObjectMap)},
511      * but instead of computing MT * M,
512      * it computes MT * C * M, where C is a diagonal matrix of 1s and 0s. This is like pretending some
513      * rows of M are 0.
514      * 
515      * @see MatrixUtils#transposeTimesSelf(com.cloudera.oryx.common.collection.LongObjectMap)
516      */
517     private static RealMatrix partialTransposeTimesSelf(LongObjectMap<float[]> M,
518                                                         int dimension, 
519                                                         LongPrimitiveIterator keys) {
520       RealMatrix result = new Array2DRowRealMatrix(dimension, dimension);
521       while (keys.hasNext()) {
522         long key = keys.next();
523         float[] vector = M.get(key);
524         for (int row = 0; row < dimension; row++) {
525           float rowValue = vector[row];
526           for (int col = 0; col < dimension; col++) {
527             result.addToEntry(row, col, rowValue * vector[col]);
528           }
529         }
530       }
531       return result;
532     }
