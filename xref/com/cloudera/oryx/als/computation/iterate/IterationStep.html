16  package com.cloudera.oryx.als.computation.iterate;
17  
18  import com.cloudera.oryx.als.computation.ALSJobStep;
19  import com.cloudera.oryx.als.computation.ALSJobStepConfig;
20  import com.cloudera.oryx.common.servcomp.Namespaces;
21  import com.cloudera.oryx.computation.common.JobStepConfig;
22  
23  /**
24   * Superclass of {@link com.cloudera.oryx.computation.common.JobStep}s that are run many times,
25   * iteratively.
26   * 
27   * @author Sean Owen
28   */
29  public abstract class IterationStep extends ALSJobStep {
30  
31    private IterationState iterationState;
32  
33    @Override
34    protected String getCustomJobName() {
35      IterationState iterationState = getIterationState();
36      StringBuilder name = new StringBuilder(100);
37      JobStepConfig config = getConfig();
38      name.append("Oryx-").append(config.getInstanceDir());
39      name.append('-').append(config.getGenerationID());
40      name.append('-').append(iterationState.getIteration());
41      name.append('-').append(iterationState.isComputingX() ? 'X' : 'Y');
42      name.append('-').append(getClass().getSimpleName());
43      return name.toString();
44    }
45  
46    protected final IterationState getIterationState() {
47      if (iterationState != null) {
48        return iterationState;
49      }
50      ALSJobStepConfig config = getConfig();
51      int currentIteration = config.getIteration();
52      String iterationsKey =
53          Namespaces.getIterationsPrefix(config.getInstanceDir(), config.getGenerationID()) +
54          currentIteration + '/';
55      iterationState = new IterationState(config.isComputingX(), currentIteration, iterationsKey);
56      return iterationState;
57    }
58    
59  }
