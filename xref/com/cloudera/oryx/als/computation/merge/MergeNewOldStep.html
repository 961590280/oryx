42  /**
43   * @author Sean Owen
44   */
45  public final class MergeNewOldStep extends ALSJobStep {
46  
47    private static final PTableType<Pair<Long, Integer>, NumericIDValue> JOIN_TYPE = Avros.tableOf(
48        Avros.pairs(ALSTypes.LONGS, ALSTypes.INTS),
49        ALSTypes.IDVALUE);
50  
51    @Override
52    protected MRPipeline createPipeline() throws IOException {
53  
54      JobStepConfig jobConfig = getConfig();
55  
56      String instanceDir = jobConfig.getInstanceDir();
57      int generationID = jobConfig.getGenerationID();
58      int lastGenerationID = jobConfig.getLastGenerationID();
59  
60      String outputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "input/";
61      if (!validOutputPath(outputKey)) {
62        return null;
63      }
64  
65      MRPipeline p = createBasicPipeline(MergeNewOldValuesFn.class);
66  
67      String trainKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "train/";
68  
69      PCollection<Pair<Long, NumericIDValue>> parsed = p.read(textInput(trainKey))
70          .parallelDo("trainParse", new DelimitedInputParseFn(),
71              Avros.pairs(ALSTypes.LONGS, ALSTypes.IDVALUE));
72  
73      PTable<Pair<Long, Integer>, NumericIDValue> train = parsed.parallelDo("train", new InboundJoinFn(), JOIN_TYPE);
74  
75      float decayFactor = (float) ConfigUtils.getDefaultConfig().getDouble("model.decay.factor");
76  
77      // Only join past data if it exists, and, decay factor is not 0
78      if (lastGenerationID >= 0 && decayFactor > 0.0f) {
79        String inputPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, lastGenerationID) + "input/";
80        Store store = Store.get();
81        Preconditions.checkState(store.exists(inputPrefix, false), "Input path does not exist: %s", inputPrefix);
82        PTable<Pair<Long, Integer>, NumericIDValue> joinInputBefore = p.read(input(inputPrefix, ALSTypes.VALUE_MATRIX))
83            .parallelDo("lastGenerationInput", new JoinBeforeMapFn(), JOIN_TYPE);
84        train = train.union(joinInputBefore);
85  
86        String testPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, lastGenerationID) + "test/";
87        if (store.exists(testPrefix, false)) {
88          // Held out test data last time; incorporate now
89          PTable<Pair<Long, Integer>, NumericIDValue> joinTestSetBefore =
90              p.read(textInput(testPrefix))
91                  .parallelDo("testParse", new DelimitedInputParseFn(), Avros.pairs(ALSTypes.LONGS, ALSTypes.IDVALUE))
92                  .parallelDo("lastGenerationTest", new JoinBeforeMapFn(), JOIN_TYPE);
93          train = train.union(joinTestSetBefore);
94        }
95      }
96  
97      // Save input size for later
98      long trainSize = train.length().getValue();
99      File trainSizeFile = File.createTempFile("trainSize", ".txt");
100     trainSizeFile.deleteOnExit();
101     Files.write(Long.toString(trainSize), trainSizeFile, StandardCharsets.UTF_8);
102     Store.get().upload(Namespaces.getTempPrefix(instanceDir, generationID) + "trainSize.txt", trainSizeFile, false);
103     IOUtils.delete(trainSizeFile);
104 
105     GroupingOptions groupingOptions = GroupingOptions.builder()
106         .partitionerClass(JoinUtils.getPartitionerClass(train.getTypeFamily()))
107         .numReducers(getNumReducers())
108         .build();
109 
110     train
111         .groupByKey(groupingOptions)
112         .parallelDo(new MergeNewOldValuesFn(), ALSTypes.VALUE_MATRIX)
113         .write(output(outputKey));
114     return p;
115   }
116 
117   public static void main(String[] args) throws Exception {
118     run(new MergeNewOldStep(), args);
119   }
120 
121 }
