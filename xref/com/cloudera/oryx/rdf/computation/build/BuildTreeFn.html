52  /**
53   * @author Sean Owen
54   */
55  public final class BuildTreeFn extends OryxReduceDoFn<Integer, Iterable<String>, String> {
56  
57    private static final Logger log = LoggerFactory.getLogger(BuildTreeFn.class);
58  
59    private static final CharMatcher NEWLINES = CharMatcher.anyOf("\n\r");
60  
61    private int numLocalTrees;
62    private int trainingFoldsPerTree;
63    private final RandomDataGenerator random = new RandomDataGenerator(RandomManager.getRandom());
64  
65    @Override
66    public void initialize() {
67      super.initialize();
68      int numReducers = getContext().getNumReduceTasks();
69      log.info("{} reducers", numReducers);
70  
71      Config config = ConfigUtils.getDefaultConfig();
72      int numTrees = config.getInt("model.num-trees");
73      // Bump this up to as least 2x reducers
74      numTrees = Math.max(numTrees, 2 * numReducers);
75      // Make it a multiple of # reducers
76      while ((numTrees % numReducers) != 0) {
77        numTrees++;
78      }
79      log.info("{} total trees", numTrees);
80  
81      Preconditions.checkArgument(numTrees % numReducers == 0, "numLocalTrees not a multiples of numReducers");
82      numLocalTrees = numTrees / numReducers;
83      Preconditions.checkArgument(numLocalTrees > 1, "Not building multiple trees per reducer");
84      log.info("Building {} trees locally", numLocalTrees);
85  
86      double sampleRate = config.getDouble("model.sample-rate");
87      Preconditions.checkArgument(sampleRate > 0.0 && sampleRate <= 1.0);
88  
89      // Arbitrarily use about 90% for training, locally
90      trainingFoldsPerTree = Math.min(numLocalTrees - 1, (int) (0.9 * numLocalTrees));
91  
92      log.info("{} training folds per tree", trainingFoldsPerTree);
93    }
94  
95    @Override
96    public void process(Pair<Integer,Iterable<String>> input, Emitter<String> emitter) {
97  
98      InboundSettings inboundSettings = InboundSettings.create(ConfigUtils.getDefaultConfig());
99      int numFeaturesAndTarget = inboundSettings.getColumnNames().size();
100     Integer targetColumn = inboundSettings.getTargetColumn();
101     Preconditions.checkNotNull(targetColumn, "No target-column specified");
102 
103     List<Example> allExamples = new ArrayList<>();
104     Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping = new HashMap<>();
105 
106     log.info("Reading input");
107     for (String line : input.second()) {
108       String[] tokens = DelimitedDataUtils.decode(line);
109       Preconditions.checkArgument(tokens.length == numFeaturesAndTarget,
110                                   "Wrong number of tokens in line: %s", line);
111       Feature target = null;
112       Feature[] features = new Feature[numFeaturesAndTarget]; // Too big by 1 but makes math easier
113       for (int col = 0; col < numFeaturesAndTarget; col++) {
114         if (col == targetColumn) {
115           target = buildFeature(col, tokens[col], inboundSettings, columnToCategoryNameToIDMapping);
116           features[col] = IgnoredFeature.INSTANCE;
117         } else {
118           features[col] = buildFeature(col, tokens[col], inboundSettings, columnToCategoryNameToIDMapping);
119         }
120       }
121       Preconditions.checkNotNull(target, "No target value in line %s", line);
122       allExamples.add(new Example(target, features));
123     }
124 
125     if (allExamples.isEmpty()) {
126       return;
127     }
128 
129     log.info("Read {} examples", allExamples.size());
130 
131     for (int treeID = 0; treeID < numLocalTrees; treeID++) {
132       List<Example> trainingExamples = new ArrayList<>();
133       List<Example> cvExamples = new ArrayList<>();
134       for (Example example : allExamples) {
135         if (random.nextInt(0, numLocalTrees - 1) < trainingFoldsPerTree) {
136           trainingExamples.add(example);
137         } else {
138           cvExamples.add(example);
139         }
140       }
141 
142       log.info("Tree {}: {} training examples", treeID, trainingExamples.size());
143       log.info("Tree {}: {} CV examples", treeID, cvExamples.size());
144       Preconditions.checkState(!trainingExamples.isEmpty(), "No training examples sampled?");
145       Preconditions.checkState(!cvExamples.isEmpty(), "No CV examples sampled?");
146 
147       progress(); // Helps prevent timeouts
148       DecisionTree tree = DecisionTree.fromExamplesWithDefault(trainingExamples);
149       progress(); // Helps prevent timeouts
150       log.info("Built tree {} with {} nodes", treeID, tree.countNodes());
151       ExampleSet cvSet = new ExampleSet(cvExamples);
152       double[] weightEval = Evaluation.evaluateToWeight(tree, cvSet);
153       progress();
154       double weight = weightEval[0];
155       double eval = weightEval[1];
156       double[] featureImportances = tree.featureImportance(cvSet);
157       progress(); // Helps prevent timeouts
158       log.info("Evaluated tree {}: {}", treeID, eval);
159       log.info("Feature importances: {}", featureImportances);
160 
161       DecisionForest singletonForest = new DecisionForest(
162           new DecisionTree[] { tree }, new double[] { weight }, featureImportances);
163 
164       String pmmlFileContents;
165       try  {
166         StringWriter treePMML = new StringWriter();
167         DecisionForestPMML.write(treePMML, singletonForest, columnToCategoryNameToIDMapping);
168         pmmlFileContents = NEWLINES.removeFrom(treePMML.toString());
169       } catch (IOException ioe) {
170         throw new IllegalStateException(ioe);
171       }
172 
173       log.info("Emitting tree {}", treeID);
174       emitter.emit(pmmlFileContents);
175       progress();
176     }
177 
178   }
179 
180   private static Feature buildFeature(int columnNumber,
181                                       String token,
182                                       InboundSettings inboundSettings,
183                                       Map<Integer, BiMap<String, Integer>> columnToCategoryNameToIDMapping) {
184     if (token == null || token.isEmpty()) {
185       return null;
186     }
187     if (inboundSettings.isNumeric(columnNumber)) {
188       return NumericFeature.forValue(Float.parseFloat(token));
189     }
190     if (inboundSettings.isCategorical(columnNumber)) {
191       return CategoricalFeature.forValue(categoricalFromString(columnNumber, token, columnToCategoryNameToIDMapping));
192     }
193     return IgnoredFeature.INSTANCE;
194   }
195 
196   private static int categoricalFromString(int columnNumber,
197                                            String value,
198                                            Map<Integer, BiMap<String, Integer>> columnToCategoryNameToIDMapping) {
199     BiMap<String,Integer> categoryNameToID = columnToCategoryNameToIDMapping.get(columnNumber);
200     if (categoryNameToID == null) {
201       categoryNameToID = HashBiMap.create();
202       columnToCategoryNameToIDMapping.put(columnNumber, categoryNameToID);
203     }
204     Integer mapped = categoryNameToID.get(value);
205     if (mapped != null) {
206       return mapped;
207     }
208     int newCategory = categoryNameToID.size();
209     categoryNameToID.put(value, newCategory);
210     return newCategory;
211   }
