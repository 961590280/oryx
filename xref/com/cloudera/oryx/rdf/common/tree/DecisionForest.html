45  /**
46   * An ensemble classifier based on many {@link DecisionTree}s.
47   *
48   * @author Sean Owen
49   * @see DecisionTree
50   */
51  public final class DecisionForest implements Iterable<DecisionTree>, TreeBasedClassifier {
52  
53    private static final Logger log = LoggerFactory.getLogger(DecisionForest.class);
54  
55    private final DecisionTree[] trees;
56    private final double[] weights;
57    private final double[] evaluations;
58    private final double[] featureImportances;
59  
60    public static DecisionForest fromExamplesWithDefault(List<Example> examples) {
61      Config config = ConfigUtils.getDefaultConfig();
62      int numTrees = config.getInt("model.num-trees");
63      double fractionOfFeaturesToTry = config.getDouble("model.fraction-features-to-try");
64      int minNodeSize = config.getInt("model.min-node-size");
65      double minInfoGainNats = config.getDouble("model.min-info-gain-nats");
66      int suggestedMaxSplitCandidates = config.getInt("model.max-split-candidates");
67      int maxDepth = config.getInt("model.max-depth");
68      double sampleRate = config.getDouble("model.sample-rate");
69      ExampleSet exampleSet = new ExampleSet(examples);
70      return new DecisionForest(numTrees,
71                                fractionOfFeaturesToTry,
72                                minNodeSize,
73                                minInfoGainNats,
74                                suggestedMaxSplitCandidates,
75                                maxDepth,
76                                sampleRate,
77                                exampleSet);
78    }
79  
80    public DecisionForest(final int numTrees,
81                          double fractionOfFeaturesToTry,
82                          final int minNodeSize,
83                          final double minInfoGainNats,
84                          final int suggestedMaxSplitCandidates,
85                          final int maxDepth,
86                          final double sampleRate,
87                          final ExampleSet examples) {
88      Preconditions.checkArgument(numTrees > 1);
89      final int numFeatures = examples.getNumFeatures();
90      Preconditions.checkArgument(fractionOfFeaturesToTry > 0.0 && fractionOfFeaturesToTry <= 1.0);
91      final int featuresToTry = Math.max(1, (int) (fractionOfFeaturesToTry * numFeatures));
92      Preconditions.checkArgument(numFeatures >= 1);
93      Preconditions.checkArgument(minNodeSize >= 1);
94      Preconditions.checkArgument(minInfoGainNats >= 0.0);
95      Preconditions.checkArgument(suggestedMaxSplitCandidates >= 1);
96      Preconditions.checkArgument(maxDepth >= 1);
97      Preconditions.checkArgument(sampleRate > 0.0 && sampleRate <= 1.0);
98  
99      weights = new double[numTrees];
100     Arrays.fill(weights, 1.0);
101     evaluations = new double[numTrees];
102     Arrays.fill(evaluations, Double.NaN);
103     final double[][] perTreeFeatureImportances = new double[numTrees][];
104 
105     // Going to set an arbitrary upper bound on the training size of about 90%
106     int maxFolds = Math.min(numTrees - 1, (int) (0.9 * numTrees));
107     // Going to set an arbitrary lower bound on the CV size of about 10%
108     int minFolds = Math.max(1, (int) (0.1 * numTrees));
109     final int folds = Math.min(maxFolds, Math.max(minFolds, (int) (sampleRate * numTrees)));
110 
111     trees = new DecisionTree[numTrees];
112 
113     ExecutorService executor = Executors.newFixedThreadPool(determineParallelism(trees.length));
114     try {
115       Collection<Future<Object>> futures = new ArrayList<>(trees.length);
116       for (int i = 0; i < numTrees; i++) {
117         final int treeID = i;
118         futures.add(executor.submit(new Callable<Object>() {
119           @Override
120           public Void call() throws Exception {
121             Collection<Example> allExamples = examples.getExamples();
122             int totalExamples = allExamples.size();
123             int expectedTrainingSize = (int) (totalExamples * sampleRate);
124             int expectedCVSize = totalExamples - expectedTrainingSize;
125             List<Example> trainingExamples = new ArrayList<>(expectedTrainingSize);
126             List<Example> cvExamples = new ArrayList<>(expectedCVSize);
127             for (Example example : allExamples) {
128               if (IntMath.mod(IntMath.mod(example.hashCode(), numTrees) - treeID, numTrees) < folds) {
129                 trainingExamples.add(example);
130               } else {
131                 cvExamples.add(example);
132               }
133             }
134 
135             Preconditions.checkState(!trainingExamples.isEmpty(), "No training examples sampled?");
136             Preconditions.checkState(!cvExamples.isEmpty(), "No CV examples sampled?");
137 
138             log.info("Starting tree {}", treeID);
139             trees[treeID] = new DecisionTree(numFeatures,
140                                              featuresToTry,
141                                              minNodeSize,
142                                              minInfoGainNats,
143                                              suggestedMaxSplitCandidates,
144                                              maxDepth,
145                                              examples.subset(trainingExamples));
146             log.info("Finished tree {}", treeID);
147             ExampleSet cvExampleSet = examples.subset(cvExamples);
148             double[] weightEval = Evaluation.evaluateToWeight(trees[treeID], cvExampleSet);
149             weights[treeID] = weightEval[0];
150             evaluations[treeID] = weightEval[1];
151             perTreeFeatureImportances[treeID] = trees[treeID].featureImportance(cvExampleSet);
152             log.info("Tree {} eval: {}", treeID, weightEval[1]);
153             return null;
154           }
155         }));
156       }
157       ExecutorUtils.getResults(futures);
158     } finally {
159       ExecutorUtils.shutdownNowAndAwait(executor);
160     }
161 
162     featureImportances = new double[numFeatures];
163     for (double[] perTreeFeatureImportance : perTreeFeatureImportances) {
164       for (int i = 0; i < numFeatures; i++) {
165         featureImportances[i] += perTreeFeatureImportance[i];
166       }
167     }
168     for (int i = 0; i < numFeatures; i++) {
169       featureImportances[i] /= numTrees;
170     }
171   }
172 
173   public DecisionForest(DecisionTree[] trees, double[] weights, double[] featureImportances) {
174     this.trees = trees;
175     this.weights = weights;
176     this.evaluations = new double[weights.length];
177     this.featureImportances = featureImportances;
178   }
179 
180   @Override
181   public Iterator<DecisionTree> iterator() {
182     return ArrayIterator.forArray(trees);
183   }
184 
185   /**
186    * @return {@link DecisionTree}s in the ensemble forest
187    */
188   public DecisionTree[] getTrees() {
189     return trees;
190   }
191 
192   public double[] getWeights() {
193     return weights;
194   }
195 
196   public double[] getEvaluations() {
197     return evaluations;
198   }
199 
200   public double[] getFeatureImportances() {
201     return featureImportances;
202   }
203   
204   @Override
205   public Prediction classify(Example test) {
206     return WeightedPrediction.voteOnFeature(
207         Lists.transform(Arrays.asList(trees), new TreeToPredictionFunction(test)), weights);
208   }
209 
210   @Override
211   public void update(Example train) {
212     for (TreeBasedClassifier tree : trees) {
213       tree.update(train);
214     }
215   }
216 
217   private static int determineParallelism(int numTrees) {
218     int numCores = ExecutorUtils.getParallelism();
219     if (numCores >= numTrees) {
220       return numTrees;
221     }
222     // Try to round up threads so trees is a multiple of it
223     int numThreads = numCores;
224     while (numTrees % numThreads != 0 && numThreads < 2 * numCores) {
225       numThreads++;
226     }
227     return numThreads;
228   }
229 
230   @Override
231   public String toString() {
232     StringBuilder result = new StringBuilder();
233     for (DecisionTree tree : trees) {
234       result.append(tree).append('\n');
235     }
236     return result.toString();
237   }
238 
239   private static final class TreeToPredictionFunction implements Function<DecisionTree, Prediction> {
240     private final Example test;
241     TreeToPredictionFunction(Example test) {
242       this.test = test;
243     }
244     @Override
245     public Prediction apply(DecisionTree tree) {
246       return tree.classify(test);
247     }
248   }
