38  /**
39   * A decision-tree classifier. Given a set of training {@link Example}s, builds a model by randomly choosing
40   * subsets of features and the training set, and then finding a binary {@link Decision} over those features and data
41   * that produces the largest information gain in the two subsets it implies. This is repeated to build a tree
42   * of {@link DecisionNode}s. At the bottom, leaf nodes are formed ({@link TerminalNode}) that contain a
43   * {@link Prediction} of the target value.
44   *
45   * @author Sean Owen
46   * @see DecisionForest
47   */
48  public final class DecisionTree implements TreeBasedClassifier {
49  
50    private final TreeNode root;
51  
52    public static DecisionTree fromExamplesWithDefault(List<Example> examples) {
53      ExampleSet exampleSet = new ExampleSet(examples);
54      Config config = ConfigUtils.getDefaultConfig();
55      int numFeatures = exampleSet.getNumFeatures();
56      double fractionOfFeaturesToTry = config.getDouble("model.fraction-features-to-try");
57      int featuresToTry = Math.max(1, (int) (fractionOfFeaturesToTry * numFeatures));
58      int minNodeSize = config.getInt("model.min-node-size");
59      double minInfoGainNats = config.getDouble("model.min-info-gain-nats");
60      int suggestedMaxSplitCandidates = config.getInt("model.max-split-candidates");
61      int maxDepth = config.getInt("model.max-depth");
62      return new DecisionTree(numFeatures,
63                              featuresToTry,
64                              minNodeSize,
65                              minInfoGainNats,
66                              suggestedMaxSplitCandidates,
67                              maxDepth,
68                              exampleSet);
69    }
70  
71    public DecisionTree(int numFeatures,
72                        int featuresToTry,
73                        int minNodeSize,
74                        double minInfoGainNats,
75                        int suggestedMaxSplitCandidates,
76                        int maxDepth,
77                        ExampleSet examples) {
78      Preconditions.checkArgument(numFeatures >= 1);
79      Preconditions.checkArgument(minNodeSize >= 1);
80      Preconditions.checkArgument(minInfoGainNats >= 0.0);
81      Preconditions.checkArgument(suggestedMaxSplitCandidates >= 1);
82      Preconditions.checkArgument(maxDepth >= 1);
83      RandomGenerator random = RandomManager.getRandom();
84      root = build(examples,
85                   1,
86                   minNodeSize,
87                   minInfoGainNats,
88                   featuresToTry,
89                   numFeatures,
90                   suggestedMaxSplitCandidates,
91                   maxDepth,
92                   random);
93    }

95    public DecisionTree(TreeNode root) {
96      Preconditions.checkNotNull(root);
97      this.root = root;
98    }

107   /**
108    * @return count of nodes in the tree
109    */
110   public int countNodes() {
111     LinkedList<TreeNode> nodes = new LinkedList<>();
112     nodes.add(root);
113     int count = 0;
114     while (!nodes.isEmpty()) {
115       count++;
116       TreeNode node = nodes.removeFirst();
117       if (!node.isTerminal()) {
118         DecisionNode decision = (DecisionNode) node;
119         nodes.add(decision.getLeft());
120         nodes.add(decision.getRight());
121       }
122     }
123     return count;
124   }

126   private static TreeNode build(ExampleSet examples,
127                                 int buildAtDepth,
128                                 int minNodeSize,
129                                 double minInfoGainNats,
130                                 int featuresToTry,
131                                 int numFeatures,
132                                 int suggestedMaxSplitCandidates,
133                                 int maxDepth,
134                                 RandomGenerator random) {
135     if (buildAtDepth >= maxDepth - 1 || examples.getExamples().size() < minNodeSize) {
136       return new TerminalNode(Prediction.buildPrediction(examples));
137     }
138 
139     double bestGain = Double.NEGATIVE_INFINITY;
140     Decision bestDecision = null;
141 
142     for (int featureNumber : randomFeatures(examples, featuresToTry, numFeatures, random)) {
143       Collection<Decision> decisions =
144           Decision.decisionsFromExamples(examples, featureNumber, suggestedMaxSplitCandidates);
145       Pair<Decision,Double> decisionAndGain = Information.bestGain(decisions, examples);
146       if (decisionAndGain != null) {
147         double gain = decisionAndGain.getSecond();
148         if (gain > bestGain) {
149           bestGain = gain;
150           bestDecision = decisionAndGain.getFirst();
151         }
152       }
153     }
154 
155     if (Double.isNaN(bestGain) || bestGain < minInfoGainNats) {
156       return new TerminalNode(Prediction.buildPrediction(examples));
157     }
158 
159     bestDecision.setInformationGain(bestGain);
160 
161     ExampleSet[] negPosSplit = examples.split(bestDecision);
162     examples = null; // For GC?
163 
164     TreeNode left = build(negPosSplit[0],
165                           buildAtDepth + 1,
166                           minNodeSize,
167                           minInfoGainNats,
168                           featuresToTry,
169                           numFeatures,
170                           suggestedMaxSplitCandidates,
171                           maxDepth,
172                           random);
173     TreeNode right = build(negPosSplit[1],
174                            buildAtDepth + 1,
175                            minNodeSize,
176                            minInfoGainNats,
177                            featuresToTry,
178                            numFeatures,
179                            suggestedMaxSplitCandidates,
180                            maxDepth,
181                            random);
182 
183     return new DecisionNode(bestDecision, left, right);
184   }
185 
186   private static Iterable<Integer> randomFeatures(ExampleSet examples,
187                                                   int featuresToTry,
188                                                   int numFeatures,
189                                                   RandomGenerator random) {
190     Collection<Integer> features = new HashSet<>(featuresToTry);
191     int max = Math.min(numFeatures, featuresToTry);
192     int attempts = 0;
193     while (features.size() < max && attempts < 2 * featuresToTry) {
194       int featureNumber = random.nextInt(numFeatures);
195       if (examples.getFeatureType(featureNumber) != FeatureType.IGNORED) {
196         features.add(featureNumber);
197       }
198       attempts++;
199     }
200     return features;
201   }

203   @Override
204   public Prediction classify(Example test) {
205     TerminalNode terminalNode = findTerminal(test);
206     return terminalNode.getPrediction();
207   }
208 
209   private TerminalNode findTerminal(Example example) {
210     TreeNode node = root;
211     while (!node.isTerminal()) {
212       DecisionNode decisionNode = (DecisionNode) node;
213       if (decisionNode.getDecision().isPositive(example)) {
214         node = decisionNode.getRight();
215       } else {
216         node = decisionNode.getLeft();
217       }
218     }
219     return (TerminalNode) node;
220   }

222   public double[] featureImportance(ExampleSet testSet) {
223     int numFeatures = testSet.getNumFeatures();
224     double[] totalGains = new double[numFeatures];
225     for (Example test : testSet) {
226       TreeNode node = root;
227       while (!node.isTerminal()) {
228         DecisionNode decisionNode = (DecisionNode) node;
229         Decision decision = decisionNode.getDecision();
230         int feature = decision.getFeatureNumber();
231         totalGains[feature] += decision.getInformationGain();
232         if (decision.isPositive(test)) {
233           node = decisionNode.getRight();
234         } else {
235           node = decisionNode.getLeft();
236         }
237       }
238     }
239     double maxGain = 0.0;
240     for (int i = 0; i < numFeatures; i++) {
241       double gain = totalGains[i];
242       Preconditions.checkState(gain >= 0.0);
243       maxGain = Math.max(maxGain, gain);
244     }
245     if (maxGain > 0.0) {
246       for (int i = 0; i < numFeatures; i++) {
247         totalGains[i] /= maxGain;
248       }
249     }
250     // Otherwise all were 0?
251     return totalGains;
252   }

254   @Override
255   public void update(Example train) {
256     TerminalNode terminalNode = findTerminal(train);
257     terminalNode.update(train);
258   }

260   @Override
261   public String toString() {
262     StringBuilder result = new StringBuilder();
263     if (root != null) {
264       Deque<Pair<TreeNode,TreePath>> toPrint = new LinkedList<>();
265       toPrint.push(new Pair<>(root, TreePath.EMPTY));
266       while (!toPrint.isEmpty()) {
267         Pair<TreeNode,TreePath> entry = toPrint.pop();
268         TreeNode node = entry.getFirst();
269         TreePath path = entry.getSecond();
270         int pathLength = path.length();
271         for (int i = 0; i < pathLength; i++) {
272           if (i == pathLength - 1) {
273             result.append(" +-");
274           } else {
275             result.append(path.isLeftAt(i) ? " | " : "   ");
276           }
277         }
278         result.append(node).append('\n');
279         if (node != null && !node.isTerminal()) {
280           DecisionNode decisionNode = (DecisionNode) node;
281           toPrint.push(new Pair<>(decisionNode.getRight(), path.extendRight()));
282           toPrint.push(new Pair<>(decisionNode.getLeft(), path.extendLeft()));
283         }
284       }
285     }
286     return result.toString();
287   }
