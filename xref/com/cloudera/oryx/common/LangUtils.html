16  package com.cloudera.oryx.common;
17  
18  import com.google.common.base.Preconditions;
19  
20  /**
21   * General utility methods related to the language, or primitives.
22   *
23   * @author Sean Owen
24   */
25  public final class LangUtils {
26  
27    private LangUtils() {
28    }
29  
30    /**
31     * Parses a {@code float} from a {@link String} as if by {@link Float#valueOf(String)}, but disallows special
32     * values like {@link Float#NaN}, {@link Float#POSITIVE_INFINITY} and {@link Float#NEGATIVE_INFINITY}.
33     *
34     * @param s {@link String} to parse
35     * @return floating-point value in the {@link String}
36     * @throws NumberFormatException if input does not parse as a floating-point value
37     * @throws IllegalArgumentException if input is infinite or {@link Float#NaN}
38     * @see #parseDouble(String)
39     */
40    public static float parseFloat(String s) {
41      float value = Float.parseFloat(s);
42      Preconditions.checkArgument(!Float.isNaN(value) && !Float.isInfinite(value), "Bad value: %s", value);
43      return value;
44    }
45  
46    /**
47     * @param s {@link String} to parse
48     * @return floating-point value in the {@link String}
49     * @see #parseFloat(String)
50     */
51    public static double parseDouble(String s) {
52      double value = Double.parseDouble(s);
53      Preconditions.checkArgument(!Double.isNaN(value) && !Double.isInfinite(value), "Bad value: %s", value);
54      return value;
55    }
56  
57    /**
58     * Like {@link com.google.common.primitives.Doubles#hashCode(double)} but avoids
59     * creating a whole new object!
60     *
61     * @param d double to hash
62     * @return the same value produced by {@link Double#hashCode()}
63     */
64    public static int hashDouble(double d) {
65      long bits = Double.doubleToLongBits(d);
66      return (int) (bits ^ (bits >>> 32));
67    }
68  
69  }
