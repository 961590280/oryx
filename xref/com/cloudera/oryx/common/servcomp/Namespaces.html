16  package com.cloudera.oryx.common.servcomp;
17  
18  import com.google.common.base.Preconditions;
19  import com.google.common.base.Strings;
20  import com.typesafe.config.Config;
21  import org.apache.hadoop.fs.FileSystem;
22  import org.apache.hadoop.fs.Path;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import java.net.URI;
27  
28  import com.cloudera.oryx.common.settings.ConfigUtils;
29  
30  /**
31   * Represents resource names on HDFS.
32   * 
33   * @author Sean Owen
34   */
35  public final class Namespaces {
36  
37    private static final Logger log = LoggerFactory.getLogger(Namespaces.class);
38  
39    private static final int DIGITS_IN_GENERATION_ID = 5;
40    public static final int MAX_GENERATION = Integer.parseInt(Strings.repeat("9", DIGITS_IN_GENERATION_ID));
41  
42    private static final Namespaces instance = new Namespaces();
43  
44    private final String prefix;
45  
46    private Namespaces() {
47      if (isLocalData()) {
48        prefix = "file:";
49      } else {
50        URI defaultURI = FileSystem.getDefaultUri(OryxConfiguration.get());
51        String host = defaultURI.getHost();
52        Preconditions.checkNotNull(host,
53            "Hadoop FS has no host in '%s'. Did you intend to set model.local-data=true?",
54            defaultURI);
55        int port = defaultURI.getPort();
56        if (port > 0) {
57          prefix = "hdfs://" + host + ':' + port;
58        } else {
59          prefix = "hdfs://" + host;
60        }
61      }
62      log.info("Namespace prefix: {}", prefix);
63    }
64  
65    /**
66     * @return singleton {@code Namespaces} instance
67     */
68    public static Namespaces get() {
69      return instance;
70    }
71  
72    public String getPrefix() {
73      return prefix;
74    }
75  
76    /**
77     * @return {@code true} if configured to use local data
78     */
79    public static boolean isLocalData() {
80      Config config = ConfigUtils.getDefaultConfig();
81      if (config.hasPath("model.local")) {
82        log.warn("model.local is deprecated; use model.local-data");
83        return config.getBoolean("model.local");
84      } else {
85        return config.getBoolean("model.local-data");
86      }
87    }
88  
89    /**
90     * @return {@code true} if configured to use local computation
91     */
92    public static boolean isLocalComputation() {
93      Config config = ConfigUtils.getDefaultConfig();
94      if (config.hasPath("model.local")) {
95        log.warn("model.local is deprecated; use model.local-computation");
96        return config.getBoolean("model.local");
97      } else {
98        return config.getBoolean("model.local-computation");
99      }
100   }
101 
102   /**
103    * @param suffix directory name
104    * @return {@link Path} appropriate for use with Hadoop representing this directory
105    */
106   public static Path toPath(String suffix) {
107     return new Path(get().getPrefix() + suffix);
108   }
109 
110   public static String getInstancePrefix(String instanceDir) {
111     return instanceDir + '/';
112   }
113 
114   public static String getInstanceGenerationPrefix(String instanceDir, int generationID) {
115     Preconditions.checkArgument(generationID >= 0L, "Bad generation %s", generationID);
116     return getInstancePrefix(instanceDir) + getPaddedGenerationID(generationID) + '/';
117   }
118 
119   private static String getPaddedGenerationID(int generationID) {
120     return Strings.padStart(Integer.toString(generationID), DIGITS_IN_GENERATION_ID, '0');
121   }
122 
123   public static String getTempPrefix(String instanceDir, int generationID) {
124     return getInstanceGenerationPrefix(instanceDir, generationID) + "tmp/";
125   }
126 
127   public static String getIterationsPrefix(String instanceDir, int generationID) {
128     return getTempPrefix(instanceDir, generationID) + "iterations/";
129   }
130 
131   public static String getGenerationDoneKey(String instanceDir, int generationID) {
132     return getInstanceGenerationPrefix(instanceDir, generationID) + "_SUCCESS";
133   }
134 
135 }
