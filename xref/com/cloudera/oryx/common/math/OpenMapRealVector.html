252   /**
253    * Optimized method to add two OpenMapRealVectors.
254    * It copies the larger vector, then iterates over the smaller.
255    *
256    * @param v Vector to add.
257    * @return the sum of {@code this} and {@code v}.
258    * @throws org.apache.commons.math3.exception.DimensionMismatchException
259    *   if the dimensions do not match.
260    */
261   public OpenMapRealVector add(OpenMapRealVector v) {
262     checkVectorDimensions(v.getDimension());
263     boolean copyThis = entries.size() > v.entries.size();
264     OpenMapRealVector res = copyThis ? this.copy() : v.copy();
265     OpenIntToDoubleHashMap.Iterator iter = copyThis ? v.entries.iterator() : entries.iterator();
266     OpenIntToDoubleHashMap randomAccess = copyThis ? entries : v.entries;
267     while (iter.hasNext()) {
268       iter.advance();
269       int key = iter.key();
270       if (randomAccess.containsKey(key)) {
271         res.setEntry(key, randomAccess.get(key) + iter.value());
272       } else {
273         res.setEntry(key, iter.value());
274       }
275     }
276     return res;
277   }

282    * @return The result of appending {@code v} to self
283    */
284   public OpenMapRealVector append(OpenMapRealVector v) {
285     OpenMapRealVector res = new OpenMapRealVector(this, v.getDimension());
286     OpenIntToDoubleHashMap.Iterator iter = v.entries.iterator();
287     while (iter.hasNext()) {
288       iter.advance();
289       res.setEntry(iter.key() + virtualSize, iter.value());
290     }
291     return res;
292   }

321   public OpenMapRealVector ebeDivide(RealVector v) {
322     checkVectorDimensions(v.getDimension());
323     OpenMapRealVector res = new OpenMapRealVector(this);
324     /*
325      * MATH-803: it is not sufficient to loop through non zero entries of
326      * this only. Indeed, if this[i] = 0d and v[i] = 0d, then
327      * this[i] / v[i] = NaN, and not 0d.
328      */
329     int n = getDimension();
330     for (int i = 0; i < n; i++) {
331       res.setEntry(i, this.getEntry(i) / v.getEntry(i));
332     }
333     return res;

337   @Deprecated
338   @Override
339   public OpenMapRealVector ebeMultiply(RealVector v) {
340     checkVectorDimensions(v.getDimension());
341     OpenMapRealVector res = new OpenMapRealVector(this);
342     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
343     while (iter.hasNext()) {
344       iter.advance();
345       res.setEntry(iter.key(), iter.value() * v.getEntry(iter.key()));
346     }
347         /*
348          * MATH-803: the above loop assumes that 0d * x  = 0d for any double x,
349          * which allows to consider only the non-zero entries of this. However,
350          * this fails if this[i] == 0d and (v[i] = NaN or v[i] = Infinity).
351          *
352          * These special cases are handled below.
353          */
354     if (v.isNaN() || v.isInfinite()) {
355       int n = getDimension();
356       for (int i = 0; i < n; i++) {
357         double y = v.getEntry(i);
358         if (Double.isNaN(y)) {
359           res.setEntry(i, Double.NaN);
360         } else if (Double.isInfinite(y)) {
361           double x = this.getEntry(i);
362           res.setEntry(i, x * y);
363         }
364       }
365     }
366     return res;
367   }

369   @Override
370   public OpenMapRealVector getSubVector(int index, int n) {
371     checkIndex(index);
372     if (n < 0) {
373       throw new NotPositiveException(LocalizedFormats.NUMBER_OF_ELEMENTS_SHOULD_BE_POSITIVE, n);
374     }
375     checkIndex(index + n - 1);
376     OpenMapRealVector res = new OpenMapRealVector(n);
377     int end = index + n;
378     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
379     while (iter.hasNext()) {
380       iter.advance();
381       int key = iter.key();
382       if (key >= index && key < end) {
383         res.setEntry(key - index, iter.value());
384       }
385     }
386     return res;
387   }

394   /**
395    * Optimized method to compute distance.
396    *
397    * @param v Vector to compute distance to.
398    * @return the distance from {@code this} and {@code v}.
399    * @throws org.apache.commons.math3.exception.DimensionMismatchException
400    *   if the dimensions do not match.
401    */
402   public double getDistance(OpenMapRealVector v) {
403     checkVectorDimensions(v.getDimension());
404     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
405     double res = 0;
406     while (iter.hasNext()) {
407       iter.advance();
408       int key = iter.key();
409       double delta = iter.value() - v.getEntry(key);
410       res += delta * delta;
411     }
412     iter = v.getEntries().iterator();
413     while (iter.hasNext()) {
414       iter.advance();
415       int key = iter.key();
416       if (!entries.containsKey(key)) {
417         double value = iter.value();
418         res += value * value;
419       }
420     }
421     return FastMath.sqrt(res);
422   }

440   /**
441    * Distance between two vectors.
442    * This method computes the distance consistent with
443    * L<sub>1</sub> norm, i.e. the sum of the absolute values of
444    * elements differences.
445    *
446    * @param v Vector to which distance is requested.
447    * @return distance between this vector and {@code v}.
448    * @throws org.apache.commons.math3.exception.DimensionMismatchException
449    *   if the dimensions do not match.
450    */
451   public double getL1Distance(OpenMapRealVector v) {
452     checkVectorDimensions(v.getDimension());
453     double max = 0;
454     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
455     while (iter.hasNext()) {
456       iter.advance();
457       double delta = FastMath.abs(iter.value() - v.getEntry(iter.key()));
458       max += delta;
459     }
460     iter = v.getEntries().iterator();
461     while (iter.hasNext()) {
462       iter.advance();
463       int key = iter.key();
464       if (!entries.containsKey(key)) {
465         double delta = FastMath.abs(iter.value());
466         max +=  FastMath.abs(delta);
467       }
468     }
469     return max;
470   }

482   /**
483    * Optimized method to compute LInfDistance.
484    *
485    * @param v Vector to compute distance from.
486    * @return the LInfDistance.
487    * @throws org.apache.commons.math3.exception.DimensionMismatchException
488    *   if the dimensions do not match.
489    */
490   private double getLInfDistance(OpenMapRealVector v) {
491     checkVectorDimensions(v.getDimension());
492     double max = 0;
493     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
494     while (iter.hasNext()) {
495       iter.advance();
496       double delta = FastMath.abs(iter.value() - v.getEntry(iter.key()));
497       if (delta > max) {
498         max = delta;
499       }
500     }
501     iter = v.getEntries().iterator();
502     while (iter.hasNext()) {
503       iter.advance();
504       int key = iter.key();
505       if (!entries.containsKey(key) && iter.value() > max) {
506         max = iter.value();
507       }
508     }
509     return max;
510   }

524     boolean infiniteFound = false;
525     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
526     while (iter.hasNext()) {
527       iter.advance();
528       double value = iter.value();
529       if (Double.isNaN(value)) {
530         return false;
531       }
532       if (Double.isInfinite(value)) {
533         infiniteFound = true;
534       }
535     }
536     return infiniteFound;
537   }

632   public void unitize() {
633     double norm = getNorm();
634     if (isDefaultValue(norm)) {
635       throw new MathArithmeticException(LocalizedFormats.ZERO_NORM);
636     }
637     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
638     while (iter.hasNext()) {
639       iter.advance();
640       entries.put(iter.key(), iter.value() / norm);
641     }
642   }

655   /**
656    * {@inheritDoc}
657    * Implementation Note: This works on exact values, and as a result
658    * it is possible for {@code a.subtract(b)} to be the zero vector, while
659    * {@code a.hashCode() != b.hashCode()}.
660    */
661   @Override
662   public int hashCode() {
663     int prime = 31;
664     int result = 1;
665     long temp = Double.doubleToLongBits(epsilon);
666     result = prime * result + (int) (temp ^ (temp >>> 32));
667     result = prime * result + virtualSize;
668     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
669     while (iter.hasNext()) {
670       iter.advance();
671       temp = Double.doubleToLongBits(iter.value());
672       result = prime * result + (int) (temp ^ (temp >>32));
673     }
674     return result;
675   }

677   /**
678    * {@inheritDoc}
679    * Implementation Note: This performs an exact comparison, and as a result
680    * it is possible for {@code a.subtract(b}} to be the zero vector, while
681    * {@code  a.equals(b) == false}.
682    */
683   @Override
684   public boolean equals(Object obj) {
685     if (this == obj) {
686       return true;
687     }
688     if (!(obj instanceof OpenMapRealVector)) {
689       return false;
690     }
691     OpenMapRealVector other = (OpenMapRealVector) obj;
692     if (virtualSize != other.virtualSize) {
693       return false;
694     }
695     if (Double.doubleToLongBits(epsilon) !=
696         Double.doubleToLongBits(other.epsilon)) {
697       return false;
698     }
699     OpenIntToDoubleHashMap.Iterator iter = entries.iterator();
700     while (iter.hasNext()) {
701       iter.advance();
702       double test = other.getEntry(iter.key());
703       if (Double.doubleToLongBits(test) != Double.doubleToLongBits(iter.value())) {
704         return false;
705       }
706     }
707     iter = other.getEntries().iterator();
708     while (iter.hasNext()) {
709       iter.advance();
710       double test = iter.value();
711       if (Double.doubleToLongBits(test) != Double.doubleToLongBits(getEntry(iter.key()))) {
712         return false;
713       }
714     }
715     return true;
716   }

738     StringBuilder sb = new StringBuilder();
739     Iterator<Entry> inOrderIter = sparseInOrderIterator();
740     while (inOrderIter.hasNext()) {
741       Entry e = inOrderIter.next();
742       sb.append(e.getIndex()).append(':').append(e.getValue()).append(' ');
743     }
744     sb.deleteCharAt(sb.length() - 1);
745     return sb.toString();
746   }

824   protected final class OpenMapSparseInOrderIterator implements Iterator<Entry> {
825 
826     private final LongPrimitiveIterator iter;
827     private final Entry current;
828 
829     OpenMapSparseInOrderIterator() {
830       List<Integer> indices = Lists.newArrayListWithCapacity(entries.size());
831       OpenIntToDoubleHashMap.Iterator it = entries.iterator();
832       while (it.hasNext()) {
833         it.advance();
834         indices.add(it.key());
835       }
836       long[] indicesArray = new long[indices.size()];
837       for (int i = 0; i < indicesArray.length; i++) {
838         indicesArray[i] = indices.get(i);
839       }
840       Arrays.sort(indicesArray);
841       this.iter = new LongPrimitiveArrayIterator(indicesArray);
842       current = new Entry();
843     }
844 
845     @Override
846     public boolean hasNext() {
847       return iter.hasNext();
848     }
849 
850     @Override
851     public Entry next() {
852       current.setIndex(iter.next().intValue());
853       return current;
854     }
855 
856     @Override
857     public void remove() {
858       throw new UnsupportedOperationException("Not supported");
859     }
860   }
