18  import java.util.List;
19  import java.util.concurrent.ExecutionException;
20  import java.util.concurrent.ExecutorService;
21  import java.util.concurrent.Executors;
22  import java.util.concurrent.Future;
23  import java.util.concurrent.ThreadFactory;
24  import java.util.concurrent.TimeUnit;

26  import com.google.common.base.Preconditions;
27  import com.google.common.collect.Lists;
28  import com.google.common.util.concurrent.ThreadFactoryBuilder;
29  import com.typesafe.config.Config;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;

33  import com.cloudera.oryx.common.settings.ConfigUtils;

35  /**
36   * Utility methods related to {@link ExecutorService} and related classes.
37   *
38   * @author Sean Owen
39   */
40  public final class ExecutorUtils {
41  
42    private static final Logger log = LoggerFactory.getLogger(ExecutorUtils.class);
43  
44    private ExecutorUtils() {
45    }
46  
47    /**
48     * @return parallelism that should be used across the project when computation is parallelized
49     */
50    public static int getParallelism() {
51      Config config = ConfigUtils.getDefaultConfig();
52      if (config.hasPath("computation-layer.parallelism")) {
53        String parallelismString = config.getString("computation-layer.parallelism");
54        if ("auto".equals(parallelismString)) {
55          return Runtime.getRuntime().availableProcessors();
56        } else {
57          int parallelism = Integer.parseInt(parallelismString);
58          Preconditions.checkArgument(parallelism >= 1, "Parallelism must be >= 1");
59          return parallelism;
60        }
61      }
62      return Runtime.getRuntime().availableProcessors();
63    }
64  
65    public static ExecutorService buildExecutor(String name) {
66      return buildExecutor(name, getParallelism());
67    }
68  
69    public static ExecutorService buildExecutor(String name, int parallelism) {
70      ThreadFactory factory = new ThreadFactoryBuilder().setNameFormat(name + "-%d").setDaemon(true).build();
71      log.info("Executing {} with parallelism {}", name, parallelism);
72      return Executors.newFixedThreadPool(parallelism, factory);
73    }
74  
75    /**
76     * Immediately shuts down its argument and waits a short time for it to terminate.
77     */
78    public static void shutdownNowAndAwait(ExecutorService executor) {
79      if (!executor.isTerminated()) {
80        if (!executor.isShutdown()) {
81          executor.shutdownNow();
82        }
83        try {
84          executor.awaitTermination(5L, TimeUnit.SECONDS);
85        } catch (InterruptedException ignored) {
86          log.warn("Interrupted while shutting down executor");
87        }
88      }
89    }
90  
91    /**
92     * Checks results for exceptions by calling all {@link Future#get()}.
93     *
94     * @throws IllegalStateException if any generated an exception
95     */
96    public static <T> List<T> getResults(Iterable<Future<T>> futures) {
97      List<T> results = Lists.newArrayList();
98      for (Future<T> future : futures) {
99        try {
100         results.add(future.get());
101       } catch (InterruptedException e) {
102         throw new IllegalStateException(e);
103       } catch (ExecutionException e) {
104         throw new IllegalStateException(e.getCause());
105       }
106     }
107     return results;
108   }
