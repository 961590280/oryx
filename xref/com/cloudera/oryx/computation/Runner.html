66  /**
67   * <p>This class will periodically run one generation of the Computation Layer using
68   * {@link com.cloudera.oryx.computation.common.DistributedGenerationRunner}.
69   * It can run after a period of time has elapsed, or an amount of
70   * data has been written.</p>
71   *
72   * <p>Example:</p>
73   *
74   * <p>{@code java -Dconfig.file=[config.conf] -jar oryx-computation-x.y.jar}</p>
75   *
76   * @author Sean Owen
77   */
78  public final class Runner implements Callable<Object>, Closeable {
79  
80    private static final Logger log = LoggerFactory.getLogger(Runner.class);
81  
82    private static final int[] ERROR_PAGE_STATUSES = {
83        HttpServletResponse.SC_BAD_REQUEST,
84        HttpServletResponse.SC_UNAUTHORIZED,
85        HttpServletResponse.SC_NOT_FOUND,
86        HttpServletResponse.SC_METHOD_NOT_ALLOWED,
87        HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
88        HttpServletResponse.SC_SERVICE_UNAVAILABLE,
89    };
90  
91    private final Config config;
92    private Tomcat tomcat;
93    private File noSuchBaseDir;
94    private boolean closed;
95  
96    public Runner() {
97      this.config = ConfigUtils.getDefaultConfig();
98    }
99  
100   /**
101    * Starts the main loop, which runs indefinitely.
102    */
103   @Override
104   public synchronized Object call() throws IOException {
105 
106     if (closed) {
107       return Boolean.FALSE;
108     }
109 
110     MemoryHandler.setSensibleLogFormat();
111     java.util.logging.Logger.getLogger("").addHandler(new MemoryHandler());
112 
113     System.setProperty("org.apache.tomcat.util.buf.UDecoder.ALLOW_ENCODED_SLASH", "true");
114 
115     this.noSuchBaseDir = Files.createTempDir();
116     this.noSuchBaseDir.deleteOnExit();
117 
118     Tomcat tomcat = new Tomcat();
119     Connector connector = makeConnector();
120     configureTomcat(tomcat, connector);
121     configureEngine(tomcat.getEngine());
122     configureServer(tomcat.getServer());
123     configureHost(tomcat.getHost());
124     Context context = makeContext(tomcat, noSuchBaseDir);
125 
126     context.addApplicationListener(ComputationInitListener.class.getName());
127 
128     addServlet(context, new PeriodicRunnerServlet(), "/periodicRunner/*");
129     addServlet(context, new style_jspx(), "/style.jspx");
130     addServlet(context, new index_jspx(), "/index.jspx");
131     addServlet(context, new status_jspx(), "/status.jspx");
132     addServlet(context, new error_jspx(), "/error.jspx");
133     addServlet(context, new LogServlet(), "/log.txt");
134 
135     try {
136       tomcat.start();
137     } catch (LifecycleException le) {
138       throw new IOException(le);
139     }
140     this.tomcat = tomcat;
141 
142     return null;
143   }
144 
145   /**
146    * Blocks and waits until the server shuts down.
147    */
148   void await() {
149     Server server = null;
150     synchronized (this) {
151       if (tomcat != null) {
152         server = tomcat.getServer();
153       }
154     }
155     if (server != null) {
156       server.await();
157     }
158   }
159 
160   @Override
161   public synchronized void close() {
162     if (!closed) {
163       closed = true;
164       if (tomcat != null) {
165         try {
166           tomcat.stop();
167           tomcat.destroy();
168         } catch (LifecycleException le) {
169           log.warn("Unexpected error while stopping", le);
170         }
171         try {
172           IOUtils.deleteRecursively(noSuchBaseDir);
173         } catch (IOException e) {
174           log.warn("Failed to delete {}", noSuchBaseDir);
175         }
176       }
177     }
178   }
179 
180   public static void main(String[] args) throws Exception {
181     try (Runner runner = new Runner()) {
182       SignalManager.register(new Runnable() {
183         @Override
184         public void run() {
185           runner.close();
186         }
187       }, SignalType.INT, SignalType.TERM);
188       runner.call();
189       runner.await();
190     }
191   }

193   private void configureTomcat(Tomcat tomcat, Connector connector) {
194     tomcat.setBaseDir(noSuchBaseDir.getAbsolutePath());
195     tomcat.setConnector(connector);
196     tomcat.getService().addConnector(connector);
197   }
198 
199   private void configureEngine(Engine engine) {
200     APISettings apiSettings = APISettings.create(config.getConfig("computation-layer.api"));
201     String userName = apiSettings.getUserName();
202     String password = apiSettings.getPassword();
203     if (userName != null && password != null) {
204       InMemoryRealm realm = new InMemoryRealm();
205       realm.addUser(userName, password);
206       engine.setRealm(realm);
207     }
208   }
209 
210   private static void configureServer(Server server) {
211     LifecycleListener jasperListener = new JasperListener();
212     server.addLifecycleListener(jasperListener);
213     jasperListener.lifecycleEvent(new LifecycleEvent(server, Lifecycle.BEFORE_INIT_EVENT, null));
214     server.addLifecycleListener(new JreMemoryLeakPreventionListener());
215     server.addLifecycleListener(new ThreadLocalLeakPreventionListener());
216   }
217 
218   private static void configureHost(Host host) {
219     host.setAutoDeploy(false);
220   }
221 
222   private Connector makeConnector() {
223     Connector connector = new Connector("org.apache.coyote.http11.Http11NioProtocol");
224     APISettings apiSettings = APISettings.create(config.getConfig("computation-layer.api"));
225     File keystoreFile = apiSettings.getKeystoreFile();
226     if (keystoreFile == null) {
227       // HTTP connector
228       connector.setPort(apiSettings.getPort());
229       connector.setSecure(false);
230       connector.setScheme("http");
231 
232     } else {
233       // HTTPS connector
234       connector.setPort(apiSettings.getSecurePort());
235       connector.setSecure(true);
236       connector.setScheme("https");
237       connector.setAttribute("SSLEnabled", "true");
238       String protocol = chooseSSLProtocol("TLSv1.2", "TLSv1.1", "TLSv1");
239       if (protocol != null) {
240         connector.setAttribute("sslProtocol", protocol);
241       }
242       connector.setAttribute("keystoreFile", keystoreFile.getAbsoluteFile());
243       connector.setAttribute("keystorePass", apiSettings.getKeystorePassword());
244     }
245 
246     // Keep quiet about the server type
247     connector.setXpoweredBy(false);
248     connector.setAttribute("server", "Oryx");
249 
250     return connector;
251   }
252   
253   private static String chooseSSLProtocol(String... protocols) {
254     for (String protocol : protocols) {
255       try {
256         SSLContext.getInstance(protocol);
257         return protocol;
258       } catch (NoSuchAlgorithmException ignored) {
259         // continue
260       }
261     }
262     return null;
263   }

265   private Context makeContext(Tomcat tomcat, File noSuchBaseDir) throws IOException {
266 
267     File contextPath = new File(noSuchBaseDir, "context");
268     IOUtils.mkdirs(contextPath);
269 
270     Context context = tomcat.addContext("", contextPath.getAbsolutePath());
271     context.setWebappVersion("3.0");
272     context.addWelcomeFile("index.jspx");
273     addErrorPages(context);
274 
275     APISettings apiSettings = APISettings.create(config.getConfig("computation-layer.api"));
276 
277     boolean needHTTPS = apiSettings.isSecure();
278     boolean needAuthentication = apiSettings.getUserName() != null;
279 
280     if (needHTTPS || needAuthentication) {
281 
282       SecurityCollection securityCollection = new SecurityCollection();
283       securityCollection.addPattern("/*");
284       SecurityConstraint securityConstraint = new SecurityConstraint();
285       securityConstraint.addCollection(securityCollection);
286 
287       if (needHTTPS) {
288         securityConstraint.setUserConstraint("CONFIDENTIAL");
289       }
290 
291       if (needAuthentication) {
292 
293         LoginConfig loginConfig = new LoginConfig();
294         loginConfig.setAuthMethod("DIGEST");
295         loginConfig.setRealmName(InMemoryRealm.NAME);
296         context.setLoginConfig(loginConfig);
297 
298         securityConstraint.addAuthRole(InMemoryRealm.AUTH_ROLE);
299 
300         context.addSecurityRole(InMemoryRealm.AUTH_ROLE);
301         context.getPipeline().addValve(new DigestAuthenticator());
302       }
303 
304       context.addConstraint(securityConstraint);
305     }
306 
307     context.setCookies(false);
308 
309     return context;
310   }

312   private static void addServlet(Context context, Servlet servlet, String path) {
313     String name = servlet.getClass().getSimpleName();
314     Tomcat.addServlet(context, name, servlet).setLoadOnStartup(1);
315     context.addServletMapping(path, name);
316   }

318   private static void addErrorPages(Context context) {
319     for (int errorCode : ERROR_PAGE_STATUSES) {
320       ErrorPage errorPage = new ErrorPage();
321       errorPage.setErrorCode(errorCode);
322       errorPage.setLocation("/error.jspx");
323       context.addErrorPage(errorPage);
324     }
325     ErrorPage errorPage = new ErrorPage();
326     errorPage.setExceptionType(Throwable.class.getName());
327     errorPage.setLocation("/error.jspx");
328     context.addErrorPage(errorPage);
329   }
