46  public final class Summarizer {
47    
48    private static final Logger LOG = LoggerFactory.getLogger(Summarizer.class);
49    
50    private final Set<Integer> ignoredColumns = new HashSet<>();
51    private final Set<Integer> categoricalColumns = new HashSet<>();
52    private Spec spec = null;
53  
54    public Summarizer spec(Spec spec) {
55      this.spec = spec;
56      return this;
57    }
58    
59    public Summarizer ignoreColumns(Integer... columns) {
60      return ignoreColumns(Arrays.asList(columns));
61    }
62  
63    public Summarizer ignoreColumns(Iterable<Integer> columns) {
64      for (Integer c : columns) {
65        ignoredColumns.add(c);
66      }
67      return this;
68    }
69  
70    public Summarizer categoricalColumns(Integer... columns) {
71      return categoricalColumns(Arrays.asList(columns));
72    }
73  
74    public Summarizer categoricalColumns(Iterable<Integer> columns) {
75      for (Integer c : columns) {
76        categoricalColumns.add(c);
77      }
78      return this;
79    }
80  
81    public PObject<Summary> build(PCollection<Record> input) {
82      return new SummaryPObject(spec, buildInternal(input));
83    }
84  
85    public PCollection<String> buildJson(PCollection<Record> input) {
86      return buildInternal(input).parallelDo("to-json", new ToJsonFn(spec), input.getTypeFamily().strings());
87    }
88  
89    private PTable<Integer, Pair<Long, InternalStats>> buildInternal(PCollection<Record> records) {
90      return records.parallelDo("summarize",
91          new SummarizeFn(ignoredColumns, categoricalColumns),
92          Avros.tableOf(Avros.ints(), Avros.pairs(Avros.longs(), Avros.reflects(InternalStats.class))))
93          .groupByKey(1)
94          .combineValues(
95              Aggregators.pairAggregator(Aggregators.SUM_LONGS(),
96                  new InternalStats.InternalStatsAggregator()));
97    }

99    private static Summary toSummary(Spec spec, Iterable<Pair<Integer, Pair<Long, InternalStats>>> iter) {
100     List<SummaryStats> ss = new ArrayList<>();
101     int fieldCount = 0;
102     long recordCount = 0L;
103     for (Pair<Integer, Pair<Long, InternalStats>> p : iter) {
104       fieldCount++;
105       recordCount = p.second().first();
106       String name = spec != null ? spec.getField(p.first()).name() : "c" + p.first();
107       SummaryStats stats = p.second().second().toSummaryStats(name, recordCount);
108       if (stats.getMissing() > 0) {
109         LOG.warn("{} missing/invalid values for numeric field {}, named '{}'",
110             stats.getMissing(), p.first(), name);
111       }
112       while (ss.size() <= p.first()) {
113         ss.add(null); // fill in any blanks
114       }
115       ss.set(p.first(), stats);
116     }
117     if (spec != null) {
118       while (ss.size() < spec.size()) {
119         ss.add(null); // Add blanks at the end
120       }
121       // Add placeholders for ignored fields in the summary
122       for (int i = 0; i < spec.size(); i++) {
123         SummaryStats s = ss.get(i);
124         if (s == null) {
125           String name = spec.getField(i).name();
126           ss.set(i, new SummaryStats(name));
127         }
128       }
129     }
130     return new Summary(recordCount, fieldCount, ss);
131   }

133   private static final class SummaryPObject extends PObjectImpl<Pair<Integer, Pair<Long, InternalStats>>, Summary> {
134     private final Spec spec;
135     
136     private SummaryPObject(Spec spec, PCollection<Pair<Integer, Pair<Long, InternalStats>>> pc) {
137       super(pc);
138       this.spec = spec;
139     }
140     
141     @Override
142     protected Summary process(Iterable<Pair<Integer, Pair<Long, InternalStats>>> iter) {
143       return toSummary(spec, iter);
144     }
145   }
146 
147   private static final class ToJsonFn extends OryxDoFn<Pair<Integer, Pair<Long, InternalStats>>, String> {
148 
149     private final Spec spec;
150     private final Collection<Pair<Integer, Pair<Long, InternalStats>>> values;
151 
152     ToJsonFn(Spec spec) {
153       this.spec = spec;
154       values = new ArrayList<>();
155     }
156 
157     @Override
158     public void process(Pair<Integer, Pair<Long, InternalStats>> input, Emitter<String> emitter) {
159       values.add(input);
160     }
161 
162     @Override
163     public void cleanup(Emitter<String> emitter) {
164       Summary summary = toSummary(spec, values);
165       ObjectMapper mapper = JacksonUtils.getObjectMapper();
166       try {
167         emitter.emit(mapper.writeValueAsString(summary));
168       } catch (IOException e) {
169         throw new CrunchRuntimeException(e);
170       }
171     }
172   }
173 
174   private static final class SummarizeFn extends OryxDoFn<Record, Pair<Integer, Pair<Long, InternalStats>>> {
175     private final Collection<Integer> ignoredColumns;
176     private final Collection<Integer> categoricalColumns;
177     private final Map<Integer, InternalStats> stats;
178     private long count;
179     
180     private SummarizeFn(Collection<Integer> ignoreColumns,
181                         Collection<Integer> categoricalColumns) {
182       this.ignoredColumns = ignoreColumns;
183       this.categoricalColumns = categoricalColumns;
184       this.stats = new HashMap<>();
185       this.count = 0;
186     }
187     
188     @Override
189     public void process(Record record,
190         Emitter<Pair<Integer, Pair<Long, InternalStats>>> emitter) {
191       for (int idx = 0; idx < record.getSpec().size(); idx++) {
192         if (!ignoredColumns.contains(idx)) {
193           InternalStats ss = stats.get(idx);
194           if (ss == null) {
195             ss = new InternalStats();
196             stats.put(idx, ss);
197           }
198           if (categoricalColumns.contains(idx)) {
199             ss.addCategorical(record.getAsString(idx));
200           } else {
201             ss.addNumeric(record.getAsDouble(idx));
202           }
203         }
204       }
205       count++;
206     }
207     
208     @Override
209     public void cleanup(Emitter<Pair<Integer, Pair<Long, InternalStats>>> emitter) {
210       for (Map.Entry<Integer, InternalStats> e : stats.entrySet()) {
211         emitter.emit(Pair.of(e.getKey(), Pair.of(count, e.getValue())));
212       }
213       stats.clear();
214     }
215   }
