87    @Override
88    public final Collection<StepState> getStepStates() throws IOException, InterruptedException {
89      String name = getCustomJobName();
90      StepStatus status = determineStatus();
91      StepState state = new StepState(startTime, endTime, name, status);
92      if (status == StepStatus.COMPLETED) {
93        // This completed, -- perhaps without ever running, which means Hadoop doesn't report state because
94        // it was skipped. It's complete.
95        state.setMapProgress(1.0f);
96        state.setReduceProgress(1.0f);
97      } else {
98        float[] progresses = determineProgresses();
99        if (progresses != null) {
100         // 0 is setup progress; we don't care
101         state.setMapProgress(progresses[1]);
102         state.setReduceProgress(progresses[2]);
103       }
104     }
105     return Collections.singletonList(state);
106   }
107 
108   private StepStatus determineStatus() throws IOException, InterruptedException {
109     JobContext job = getJob();
110     if (job == null) {
111       return StepStatus.COMPLETED;
112     }
113     Cluster cluster = new Cluster(getConf());
114     try {
115       JobID jobID = job.getJobID();
116       if (jobID == null) {
117         return StepStatus.PENDING;
118       }
119       Job runningJob = cluster.getJob(jobID);
120       if (runningJob == null) {
121         return StepStatus.PENDING;
122       }
123       JobStatus.State state = runningJob.getJobState();
124       switch (state) {
125         case PREP:
126           return StepStatus.PENDING;
127         case RUNNING:
128           return StepStatus.RUNNING;
129         case SUCCEEDED:
130           return StepStatus.COMPLETED;
131         case FAILED:
132           return StepStatus.FAILED;
133         case KILLED:
134           return StepStatus.CANCELLED;
135       }
136       throw new IllegalArgumentException("Unknown Hadoop job state " + state);
137     } finally {
138       cluster.close();
139     }
140   }

142   private JobContext getJob() {
143     return exec == null ? null : exec.getJobs().get(0).getJob();
144   }

146   /**
147    * @return three progress values, in [0,1], as a {@code float[]}, representing setup, mapper and reducer progress
148    */
149   private float[] determineProgresses() throws IOException, InterruptedException {
150     if (exec == null) {
151       return null;
152     }
153     Cluster cluster = new Cluster(getConf());
154     try {
155       JobID jobID = getJob().getJobID();
156       if (jobID == null) {
157         return null;
158       }
159       Job runningJob = cluster.getJob(jobID);
160       if (runningJob == null) {
161         return null;
162       }
163 
164       return new float[] { runningJob.setupProgress(), runningJob.mapProgress(), runningJob.reduceProgress() };
165     } finally {
166       cluster.close();
167     }
168   }

170   @Override
171   public final int run(String[] args) throws InterruptedException, IOException, JobException {
172 
173     MemoryHandler.setSensibleLogFormat();
174 
175     if (log.isInfoEnabled()) {
176       log.info("args are {}", Arrays.toString(args));
177     }
178 
179     config = parseConfig(args);
180 
181     String name = getCustomJobName();
182     log.info("Running {}", name);
183 
184     long start = System.currentTimeMillis();
185     startTime = new Date(start);
186     MRPipeline pipeline = createPipeline();
187     try {
188       if (pipeline == null) {
189         log.info("{} does not need to run", name);
190       } else {
191 
192         exec = pipeline.runAsync();
193         log.info("Waiting for {} to complete", name);
194         exec.waitUntilDone();
195 
196         PipelineExecution.Status exitStatus = exec.getStatus();
197         if (exitStatus == PipelineExecution.Status.KILLED || exitStatus == PipelineExecution.Status.FAILED) {
198           exec.kill();
199           throw new JobException(name + " failed in state " + exitStatus);
200         }
201 
202         log.info("Finished {}", name);
203         postRun();
204 
205       }
206     } catch (InterruptedException ie) {
207       log.warn("Interrupted {}", name);
208       exec.kill();
209       throw ie;
210     } finally {
211       long end = System.currentTimeMillis();
212       endTime = new Date(end);
213       log.info("Completed {} in {}s", this, (end - start) / 1000L);
214       if (pipeline != null) {
215         pipeline.done();
216       }
217     }
218 
219     return 0;
220   }

243   /**
244    * Creates a new {@link MRPipeline} instance that contains common configuration
245    * settings.
246    *
247    * @return a new {@link MRPipeline} instance, suitably configured
248    */
249   protected final MRPipeline createBasicPipeline(Class<?> jarClass) throws IOException {
250     Configuration conf = OryxConfiguration.get(getConf());
251 
252     conf.setBoolean(MRJobConfig.MAP_OUTPUT_COMPRESS, true);
253     conf.setClass(MRJobConfig.MAP_OUTPUT_COMPRESS_CODEC, SnappyCodec.class, CompressionCodec.class);
254 
255     conf.setBoolean("mapred.output.compress", true);
256     conf.set("mapred.output.compression.type", "BLOCK");
257     conf.setClass("mapred.output.compression.codec", SnappyCodec.class, CompressionCodec.class);
258     // Set old-style equivalents for Avro/Crunch's benefit
259     conf.set("avro.output.codec", "snappy");
260 
261     conf.setBoolean(MRJobConfig.MAP_SPECULATIVE, true);
262     conf.setBoolean(MRJobConfig.REDUCE_SPECULATIVE, true);
263     conf.setBoolean(TTConfig.TT_OUTOFBAND_HEARBEAT, true);
264     conf.setInt(MRJobConfig.JVM_NUMTASKS_TORUN, -1);
265 
266     //conf.setBoolean("crunch.disable.deep.copy", true);
267     // Giving one mapper a lot of data can cause issues in some stages, so default to disable this
268     conf.setBoolean("crunch.disable.combine.file", true);
269 
270     Config appConfig = ConfigUtils.getDefaultConfig();
271 
272     conf.set("crunch.tmp.dir", appConfig.getString("computation-layer.tmp-dir"));
273 
274     int mapMemoryMB = appConfig.getInt("computation-layer.mapper-memory-mb");
275     log.info("Mapper memory: {}", mapMemoryMB);
276     int mapHeapMB = (int) (mapMemoryMB / 1.3); // Matches Hadoop's default
277     log.info("Mappers have {}MB heap and can access {}MB RAM", mapHeapMB, mapMemoryMB);
278     if (conf.get(MRJobConfig.MAP_JAVA_OPTS) != null) {
279       log.info("Overriding previous setting of {}, which was '{}'",
280                MRJobConfig.MAP_JAVA_OPTS,
281                conf.get(MRJobConfig.MAP_JAVA_OPTS));
282     }
283     conf.set(MRJobConfig.MAP_JAVA_OPTS,
284              "-Xmx" + mapHeapMB + "m -XX:+UseCompressedOops -XX:+UseParallelGC -XX:+UseParallelOldGC");
285     log.info("Set {} to '{}'", MRJobConfig.MAP_JAVA_OPTS, conf.get(MRJobConfig.MAP_JAVA_OPTS));
286     // See comment below on CM
287     conf.setInt("mapreduce.map.java.opts.max.heap", mapHeapMB);
288 
289     int reduceMemoryMB = appConfig.getInt("computation-layer.reducer-memory-mb");
290     log.info("Reducer memory: {}", reduceMemoryMB);
291     if (isHighMemoryStep()) {
292       reduceMemoryMB *= appConfig.getInt("computation-layer.worker-high-memory-factor");
293       log.info("Increasing {} to {} for high-memory step", MRJobConfig.REDUCE_MEMORY_MB, reduceMemoryMB);
294     }
295     conf.setInt(MRJobConfig.REDUCE_MEMORY_MB, reduceMemoryMB);
296 
297     int reduceHeapMB = (int) (reduceMemoryMB / 1.3); // Matches Hadoop's default
298     log.info("Reducers have {}MB heap and can access {}MB RAM", reduceHeapMB, reduceMemoryMB);
299     if (conf.get(MRJobConfig.REDUCE_JAVA_OPTS) != null) {
300       log.info("Overriding previous setting of {}, which was '{}'",
301                MRJobConfig.REDUCE_JAVA_OPTS,
302                conf.get(MRJobConfig.REDUCE_JAVA_OPTS));
303     }
304     conf.set(MRJobConfig.REDUCE_JAVA_OPTS,
305              "-Xmx" + reduceHeapMB + "m -XX:+UseCompressedOops -XX:+UseParallelGC -XX:+UseParallelOldGC");
306     log.info("Set {} to '{}'", MRJobConfig.REDUCE_JAVA_OPTS, conf.get(MRJobConfig.REDUCE_JAVA_OPTS));
307     // I see this in CM but not in Hadoop docs; probably won't hurt as it's supposed to result in
308     // -Xmx appended to opts above, which is at worst redundant
309     conf.setInt("mapreduce.reduce.java.opts.max.heap", reduceHeapMB);
310 
311     // Pass total config state
312     conf.set(CONFIG_SERIALIZATION_KEY, ConfigUtils.getDefaultConfig().root().render());
313 
314     // Make sure to set any args to conf above this line!
315 
316     setConf(conf);
317 
318     Job job = Job.getInstance(conf);
319 
320     // Basic File IO settings
321     FileInputFormat.setMaxInputSplitSize(job, 1L << 28); // ~268MB
322     SequenceFileOutputFormat.setOutputCompressionType(job, SequenceFile.CompressionType.BLOCK);
323     FileOutputFormat.setCompressOutput(job, true);
324     FileOutputFormat.setOutputCompressorClass(job, SnappyCodec.class);
325 
326     log.info("Created pipeline configuration {}", job.getConfiguration());
327 
328     return new MRPipeline(jarClass, getCustomJobName(), job.getConfiguration());
329   }

331   protected final boolean validOutputPath(String outputPathKey) throws IOException {
332     Preconditions.checkArgument(outputPathKey != null && outputPathKey.endsWith("/"),
333         "%s should end with /", outputPathKey);
334 
335     String jobName = getCustomJobName();
336 
337     Store store = Store.get();
338     if (store.exists(outputPathKey, false)) {
339       log.info("Output path {} already exists", outputPathKey);
340       if (store.exists(outputPathKey + "_SUCCESS", true)) {
341         log.info("{} shows _SUCCESS present; skipping", jobName);
342         return false;
343       }
344       log.info("{} seems to have stopped during an earlier run; deleting output at {} and recomputing",
345           jobName, outputPathKey);
346       store.recursiveDelete(outputPathKey);
347     } else {
348       log.info("Output path is clear for writing: {}", outputPathKey);
349     }
350     return true;
351   }

396   protected final GroupingOptions groupingOptions(
397       Class<? extends Partitioner> partitionerClass,
398       Class<? extends RawComparator<?>> groupingComparator,
399       Class<? extends RawComparator<?>> sortComparator) {
400     GroupingOptions.Builder b = GroupingOptions.builder()
401         .partitionerClass(partitionerClass)
402         .numReducers(getNumReducers());
403 
404     if (groupingComparator != null) {
405       b.groupingComparatorClass(groupingComparator);
406     }
407     if (sortComparator != null) {
408       b.sortComparatorClass(sortComparator);
409     }
410     return b.build();
411   }

413   protected final int getNumReducers() {
414     String parallelismString = ConfigUtils.getDefaultConfig().getString("computation-layer.parallelism");
415     if ("auto".equals(parallelismString)) {
416       return 11; // Default to a prime. Helps avoid problems with funny distributions of IDs.
417     } else {
418       return Integer.parseInt(parallelismString);
419     }
420   }

422   public static void run(Tool step, String[] args) throws IOException, InterruptedException, JobException {
423     log.info("Running step {}", step.getClass().getSimpleName());
424     try {
425       int result = ToolRunner.run(OryxConfiguration.get(), step, args);
426       if (result != 0) {
427         throw new JobException(step + " hasd bad exit status: " + result);
428       }
429     } catch (IOException | JobException | InterruptedException e) {
430       throw e;
431     } catch (Exception e) {
432       throw new JobException(e);
433     }
434   }
