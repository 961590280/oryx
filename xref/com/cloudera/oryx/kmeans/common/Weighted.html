16  package com.cloudera.oryx.kmeans.common;
17  
18  import java.util.ArrayList;
19  import java.util.Collection;
20  import java.util.List;
21  import java.util.SortedMap;
22  import java.util.TreeMap;
23  
24  import com.google.common.base.Function;
25  import com.google.common.base.Preconditions;
26  import com.google.common.collect.Collections2;
27  import com.google.common.collect.Lists;
28  import org.apache.commons.math3.random.RandomGenerator;
29  
30  import com.cloudera.oryx.common.LangUtils;
31  import com.cloudera.oryx.common.random.RandomManager;
32  
33  /**
34   * A typed container for an object and an associated numeric weight.
35   */
36  public class Weighted<T> {
37  
38    private final T thing;
39    private final double weight;
40    
41    private static final class WeightFunction<T> implements Function<T, Weighted<T>> {
42      @Override
43      public Weighted<T> apply(T input) {
44        return new Weighted<>(input);
45      }
46    }
47    
48    /**
49     * Converts an input {@code Collection<T>} into a {@code Collection<Weighted>}.
50     * 
51     * @param things The items to convert
52     * @return The items as {@code Weighted<T>} values with weight 1.0
53     */
54    public static <T> Collection<Weighted<T>> create(Collection<T> things) {
55      return Collections2.transform(things, new WeightFunction<T>());
56    }
57    
58    /**
59     * Converts an input {@code List<T>} into a {@code List<Weighted>}.
60     * 
61     * @param things The items to convert
62     * @return The items as {@code Weighted<T>} values with weight 1.0
63     */
64    public static <T> List<Weighted<T>> create(List<T> things) {
65      return Lists.transform(things, new WeightFunction<T>());
66    }
67    
68    /**
69     * Sample items from a {@code List<Weighted>} where items with higher weights
70     * have a higher probability of being included in the sample.
71     * 
72     * @param things The iterable to sample from
73     * @param size The number of items to sample
74     * @return A list containing the sampled items
75     */
76    public static <T extends Weighted<?>> List<T> sample(Iterable<T> things, int size, RandomGenerator random) {
77      if (random == null) {
78        random = RandomManager.getRandom();
79      }
80      SortedMap<Double, T> sampled = new TreeMap<>();
81      for (T thing : things) {
82        if (thing.weight() > 0) {
83          double score = Math.log(random.nextDouble()) / thing.weight();
84          if (sampled.size() < size || score > sampled.firstKey()) {
85            sampled.put(score, thing);
86          }
87          if (sampled.size() > size) {
88            sampled.remove(sampled.firstKey());
89          }
90        }
91      }
92      return new ArrayList<>(sampled.values());
93    }
94  
95    public Weighted() {
96      this(null); // Mainly exists for serialization
97    }
98  
99    /**
100    * Create a new instance with weight 1.0.
101    * 
102    * @param thing The thing that is weighted
103    */
104   public Weighted(T thing) {
105     this(thing, 1.0);
106   }
107   
108   /**
109    * Create a new instance with the given weight.
110    * 
111    * @param thing The (non-null) thing to weight
112    * @param weight The weight
113    */
114   public Weighted(T thing, double weight) {
115     this.thing = Preconditions.checkNotNull(thing);
116     this.weight = weight;
117   }
118   
119   /**
120    * Return the thing referenced by this instance.
121    */
122   public T thing() {
123     return thing;
124   }
125   
126   /**
127    * Return the numeric weight for this instance.
128    */
129   public double weight() {
130     return weight;
131   }
132   
133   @Override
134   public boolean equals(Object other) {
135     if (!(other instanceof Weighted)) {
136       return false;
137     }
138     Weighted<?> wv = (Weighted<?>) other;
139     return thing.equals(wv.thing) && weight == wv.weight;
140   }
141   
142   @Override
143   public int hashCode() {
144     return 17 * thing.hashCode() + 37 * LangUtils.hashDouble(weight);
145   }
146   
147   @Override
148   public String toString() {
149     return String.valueOf(thing) + ';' + weight;
150   }
151 }
