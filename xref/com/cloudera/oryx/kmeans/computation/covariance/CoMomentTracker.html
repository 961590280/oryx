15  package com.cloudera.oryx.kmeans.computation.covariance;
16  
17  import org.apache.commons.math3.linear.RealVector;
18  
19  import java.util.HashMap;
20  import java.util.Map;
21  import java.util.Set;
22  
23  final class CoMomentTracker {
24    private final Map<Index, CoMoment> cache = new HashMap<>();
25  
26    public void reset() {
27      cache.clear();
28    }
29  
30    public void update(RealVector v) {
31      for (int i = 0; i < v.getDimension(); i++) {
32        double vi = v.getEntry(i);
33        for (int j = i; j < v.getDimension(); j++) {
34          Index idx = new Index(i, j);
35          CoMoment cm = cache.get(idx);
36          if (cm == null) {
37            cm = new CoMoment();
38            cache.put(idx, cm);
39          }
40          cm.update(vi, v.getEntry(j));
41        }
42      }
43    }
44  
45    public Set<Map.Entry<Index, CoMoment>> entrySet() {
46      return cache.entrySet();
47    }
48  }
