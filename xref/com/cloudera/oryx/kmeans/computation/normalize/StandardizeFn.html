16  package com.cloudera.oryx.kmeans.computation.normalize;
17  
18  import com.cloudera.oryx.common.math.NamedRealVector;
19  import com.cloudera.oryx.common.math.Vectors;
20  import com.cloudera.oryx.common.servcomp.Store;
21  import com.cloudera.oryx.computation.common.fn.OryxDoFn;
22  import com.cloudera.oryx.computation.common.json.JacksonUtils;
23  import com.cloudera.oryx.computation.common.records.Record;
24  import com.cloudera.oryx.computation.common.records.vectors.VectorRecord;
25  import com.cloudera.oryx.computation.common.summary.Summary;
26  import com.cloudera.oryx.computation.common.summary.SummaryStats;
27  
28  import com.google.common.base.Preconditions;
29  import org.apache.commons.math3.linear.ArrayRealVector;
30  import org.apache.commons.math3.linear.RealVector;
31  import org.apache.crunch.CrunchRuntimeException;
32  import org.apache.crunch.Emitter;
33  
34  import java.io.IOException;
35  import java.util.Collection;
36  import java.util.Collections;
37  import java.util.List;
38  
39  final class StandardizeFn extends OryxDoFn<Record, RealVector> {
40  
41    private final String summaryKey;
42    private final NormalizeSettings settings;
43    private final Collection<Integer> ignoredColumns;
44    private final Collection<Integer> idColumns;
45  
46    private Summary summary;
47    private int expansion;
48    private boolean sparse;
49  
50    StandardizeFn() {
51      this(new NormalizeSettings(),
52           Collections.<Integer>emptyList(),
53           Collections.<Integer>emptyList(),
54           "");
55    }
56  
57    StandardizeFn(Summary summary) {
58      this(new NormalizeSettings(),
59           Collections.<Integer>emptyList(),
60           Collections.<Integer>emptyList(),
61           "");
62      this.summary = summary;
63    }
64  
65    StandardizeFn(Summary summary, Transform defaultTransform) {
66      this(new NormalizeSettings(defaultTransform),
67           Collections.<Integer>emptyList(),
68           Collections.<Integer>emptyList(),
69           "");
70      this.summary = summary;
71    }
72  
73    StandardizeFn(NormalizeSettings settings,
74                  Collection<Integer> ignoredColumns,
75                  Collection<Integer> idColumns,
76                  String summaryKey) {
77      this.settings = settings;
78      this.summaryKey = summaryKey;
79      this.ignoredColumns = ignoredColumns;
80      this.idColumns = idColumns;
81    }
82  
83    @Override
84    public void initialize() {
85      super.initialize();
86      if (summaryKey != null && !summaryKey.isEmpty()) {
87        try {
88          Store store = Store.get();
89          List<String> pieces = store.list(summaryKey, true);
90          if (pieces.isEmpty()) {
91            // No summary file specified, which is okay for us
92            summary = new Summary();
93          } else if (pieces.size() > 1) {
94            throw new IllegalStateException("Expected exactly one summary file: " + pieces);
95          } else {
96            try {
97              summary = JacksonUtils.getObjectMapper().readValue(store.readFrom(pieces.get(0)), Summary.class);
98              Preconditions.checkNotNull(summary);
99            } catch (Exception e) {
100             throw new CrunchRuntimeException(e);
101           }
102         }
103       } catch (IOException e) {
104         throw new CrunchRuntimeException(e);
105       }
106     } else if (summary == null) {
107       summary = new Summary();
108     }
109 
110     //TODO: arbitrary ID columns
111     this.expansion = -ignoredColumns.size() + summary.getNetLevels() -
112         (!idColumns.isEmpty() && !ignoredColumns.contains(idColumns.iterator().next()) ? 1 : 0);
113     this.sparse = settings.getSparse() != null ? settings.getSparse() :
114         expansion > 2 * (summary.getFieldCount() - ignoredColumns.size());
115   }
116 
117   @Override
118   public void process(Record record, Emitter<RealVector> emitter) {
119     int len = record.getSpec().size() + expansion;
120     RealVector v;
121     if (record instanceof VectorRecord) {
122       RealVector innerRecord = ((VectorRecord) record).getVector();
123       if (innerRecord instanceof ArrayRealVector) {
124         v = Vectors.dense(len);
125       } else {
126         v = Vectors.sparse(len);
127       }
128     } else if (sparse) {
129       v = Vectors.sparse(len);
130     } else {
131       v = Vectors.dense(len);
132     }
133     increment("NORMALIZE", "INPUT_RECORDS");
134 
135     int offset = 0;
136     for (int i = 0; i < record.getSpec().size(); i++) {
137       if (!idColumns.contains(i) && !ignoredColumns.contains(i)) {
138         SummaryStats ss = summary.getStats(i);
139         if (ss == null || ss.isEmpty()) {
140           v.setEntry(offset, record.getAsDouble(i));
141           offset++;
142         } else if (ss.isNumeric()) {
143           Transform t = settings.getTransform(i);
144           double raw = record.getAsDouble(i);
145           if (Double.isNaN(raw)) {
146             increment("NORMALIZE", "NaN record value");
147             return;
148           }
149           double n = t.apply(raw, ss) * settings.getScale(i);
150           v.setEntry(offset, n);
151           offset++;
152         } else {
153           int index = ss.index(record.getAsString(i));
154           if (index < 0) {
155             increment("NORMALIZE", "Negative record index");
156             return;
157           }
158           v.setEntry(offset + index, settings.getScale(i));
159           offset += ss.numLevels();
160         }
161       }
162     }
163 
164     increment("NORMALIZE", "PRE_NAMING");
165     if (!idColumns.isEmpty()) {
166       // TODO: arbitrary properties
167       v = new NamedRealVector(v, record.getAsString(idColumns.iterator().next()));
168     }
169     emitter.emit(v);
170     increment("NORMALIZE", "OUTPUT_VECTORS");
171   }
172 }
