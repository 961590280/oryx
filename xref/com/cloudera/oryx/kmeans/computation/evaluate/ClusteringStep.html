55    private static final Logger log = LoggerFactory.getLogger(ClusteringStep.class);
56  
57    @Override
58    protected MRPipeline createPipeline() throws IOException {
59      JobStepConfig stepConfig = getConfig();
60      Config config = ConfigUtils.getDefaultConfig();
61      EvaluationSettings settings = EvaluationSettings.create(config);
62  
63      String instanceDir = stepConfig.getInstanceDir();
64      int generationID = stepConfig.getGenerationID();
65      String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
66      String outputKey = prefix + "eval/";
67      if (!validOutputPath(outputKey)) {
68        return null;
69      }
70      String inputKey = prefix + "weighted/weightedKSketchVectors/";
71      MRPipeline p = createBasicPipeline(ClosestSketchVectorFn.class);
72  
73      PType<Pair<Integer, WeightedRealVector>> inputType = KMeansTypes.FOLD_WEIGHTED_VECTOR;
74      PCollection<Pair<Integer, WeightedRealVector>> weightedSketchVectors = p.read(avroInput(inputKey, inputType));
75  
76      PCollection<KMeansEvaluationData> evaluation = weightedSketchVectors
77          .parallelDo("replicate",
78              new ReplicateValuesFn<Pair<Integer, WeightedRealVector>>(settings.getKValues(), settings.getReplications()),
79              Avros.tableOf(Avros.pairs(Avros.ints(), Avros.ints()), Avros.pairs(Avros.ints(), MLAvros.weightedVector())))
80          .groupByKey(settings.getParallelism())
81          .parallelDo("cluster",
82              new KMeansClusteringFn(settings),
83              Serializables.avro(KMeansEvaluationData.class));
84  
85      // Write out the centers themselves to a text file
86      evaluation.parallelDo("replicaCenters", new CentersOutputFn(prefix), Avros.strings())
87          .write(compressedTextOutput(p.getConfiguration(), outputKey + "replicaCenters/"));
88  
89      // Write out the per-replica stats
90      evaluation.parallelDo("replicaStats", new StatsOutputFn(), Avros.strings())
91          .write(compressedTextOutput(p.getConfiguration(), outputKey + "replicaStats/"));
92  
93      return p;
94    }
95  
96    @Override
97    protected void postRun() throws IOException {
98      JobStepConfig stepConfig = getConfig();
99      String instanceDir = stepConfig.getInstanceDir();
100     int generationID = stepConfig.getGenerationID();
101     String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
102     String replicaStatsKey = prefix + "eval/replicaStats/";
103     String replicaCentersKey = prefix + "eval/replicaCenters/";
104 
105     boolean succeeded = false;
106     try {
107       List<ClusterValidityStatistics> stats = findBest(replicaStatsKey);
108       uploadCenters(prefix, replicaCentersKey, stats);
109       succeeded = true;
110     } finally {
111       if (!succeeded) {
112         // This causes a re-run of the M/R next time if failed
113         Store.get().delete(prefix + "_SUCCESS");
114       }
115     }
116   }
117 
118   private static List<ClusterValidityStatistics> findBest(String replicaStatsKey) throws IOException {
119     KMeansEvalStrategy evalStrategy = EvaluationSettings.create(ConfigUtils.getDefaultConfig()).getEvalStrategy();
120     Store store = Store.get();
121     List<ClusterValidityStatistics> stats = new ArrayList<>();
122     for (String statsKey : store.list(replicaStatsKey, true)) {
123       for (String line : new FileLineIterable(store.readFrom(statsKey))) {
124         ClusterValidityStatistics cvs = ClusterValidityStatistics.parse(line);
125         log.info("PS: {}", cvs);
126         stats.add(cvs);
127       }
128     }
129     return evalStrategy == null ? stats : evalStrategy.evaluate(stats);
130   }
131 
132   private static void uploadCenters(String prefix,
133                                     String replicaCentersKey,
134                                     List<ClusterValidityStatistics> bestStats) throws IOException {
135     Store store = Store.get();
136     List<Model> models = new ArrayList<>();
137     String bestName = null;
138     if (bestStats.size() == 1) {
139       ClusterValidityStatistics cvs = bestStats.get(0);
140       bestName = prefix + ':' + cvs.getK() + ':' + cvs.getReplica();
141       log.info("Best scoring model = {}", bestName);
142     }
143     log.info("Reading model centers from key = {}", replicaCentersKey);
144     DataDictionary dictionary = null;
145     for (String modelKey : store.list(replicaCentersKey, true)) {
146       try {
147         PMML pmml;
148         try (InputStream in = store.streamFrom(modelKey)) {
149           pmml = KMeansPMML.read(in);
150         }
151         log.info("Read {} from key = {}", pmml.getModels().size(), modelKey);
152         if (dictionary == null) {
153           dictionary = pmml.getDataDictionary();
154         }
155         if (bestName == null) {
156           models.addAll(pmml.getModels());
157         } else {
158           for (Model m : pmml.getModels()) {
159             if (bestName.equals(m.getModelName())) {
160               models = ImmutableList.of(m);
161               break;
162             }
163           }
164         }
165       } catch (JAXBException | SAXException e) {
166         log.error("Serialization error", e);
167       }
168     }
169 
170     File keptModels = File.createTempFile("model", ".pmml.gz");
171     keptModels.deleteOnExit();
172     log.info("Writing {} models to model.pmml.gz...", models.size());
173     KMeansPMML.write(keptModels, dictionary, models);
174     store.upload(prefix + "model.pmml.gz", keptModels, true);
175     IOUtils.delete(keptModels);
176   }
