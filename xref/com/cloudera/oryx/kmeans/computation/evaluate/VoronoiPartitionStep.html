25  import com.typesafe.config.Config;
26  import org.apache.crunch.PCollection;
27  import org.apache.crunch.impl.mr.MRPipeline;
28  import org.apache.crunch.lib.PTables;
29  import org.apache.crunch.types.avro.Avros;
30  
31  import java.io.IOException;
32  
33  public final class VoronoiPartitionStep extends KMeansJobStep {
34  
35    @Override
36    protected MRPipeline createPipeline() throws IOException {
37      JobStepConfig stepConfig = getConfig();
38      Config config = ConfigUtils.getDefaultConfig();
39      ClusterSettings clusterSettings = ClusterSettings.create(config);
40  
41      String instanceDir = stepConfig.getInstanceDir();
42      int generationID = stepConfig.getGenerationID();
43      String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
44      String outputKey = prefix + "weighted/";
45      if (!validOutputPath(outputKey)) {
46        return null;
47      }
48  
49      String indexKey = prefix + "sketch/" + clusterSettings.getSketchIterations();
50      String inputKey = prefix + "normalized/";
51      MRPipeline p = createBasicPipeline(ClosestSketchVectorFn.class);
52  
53      // first I compute the weight of each k-sketch vector, i.e., Voronoi partition
54      // I aggregate all together and persist on disk
55      // PCollection<ClosestSketchVectorData> weights = inputPairs(p, inputKey, MLAvros.vector())
56      PCollection<ClosestSketchVectorData> weights = PTables.asPTable(inputPairs(p, inputKey, MLAvros.vector())
57          .parallelDo("computingSketchVectorWeights",
58                      new ClosestSketchVectorFn<>(indexKey, clusterSettings),
59              Avros.pairs(Avros.ints(), Avros.reflects(ClosestSketchVectorData.class))))
60          .groupByKey(1)
61          .combineValues(new ClosestSketchVectorAggregator(clusterSettings))
62          .values()
63          .write(avroOutput(outputKey + "kSketchVectorWeights/"));
64  
65      // this "pipeline" takes a single ClosestSketchVectorData and returns weighted vectors
66      // could be done outside MapReduce, but that would require me to materialize the ClosestSketchVectorData
67      weights.parallelDo(
68          "generatingWeightedSketchVectors",
69          new WeightVectorsFn(indexKey),
70          KMeansTypes.FOLD_WEIGHTED_VECTOR)
71          .write(avroOutput(outputKey + "weightedKSketchVectors/"));
72  
73      return p;
74    }
75  }
