16  package com.cloudera.oryx.kmeans.computation.local;
17  
18  import com.cloudera.oryx.common.parallel.ExecutorUtils;
19  import com.cloudera.oryx.common.random.RandomManager;
20  import com.cloudera.oryx.common.settings.ConfigUtils;
21  import com.cloudera.oryx.kmeans.common.WeightedRealVector;
22  import com.cloudera.oryx.kmeans.computation.cluster.ClusterSettings;
23  import com.cloudera.oryx.kmeans.computation.cluster.KSketchIndex;
24  import com.typesafe.config.Config;
25  import org.apache.commons.math3.linear.RealVector;
26  import org.apache.commons.math3.random.RandomGenerator;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.util.ArrayList;
31  import java.util.Collection;
32  import java.util.List;
33  import java.util.concurrent.Callable;
34  import java.util.concurrent.ExecutorService;
35  import java.util.concurrent.Future;
36  
37  public final class WeightedPointsByFold implements Callable<List<List<WeightedRealVector>>> {
38  
39    private static final Logger log = LoggerFactory.getLogger(WeightedPointsByFold.class);
40  
41    private final List<List<RealVector>> foldVecs;
42  
43    public WeightedPointsByFold(List<List<RealVector>> foldVecs) {
44      this.foldVecs = foldVecs;
45    }
46  
47    @Override
48    public List<List<WeightedRealVector>> call() {
49      Config config = ConfigUtils.getDefaultConfig();
50      ClusterSettings cluster = ClusterSettings.create(config);
51      KSketchIndex index = buildIndex(foldVecs, cluster);
52      int pointsPerIteration = cluster.getSketchPoints();
53      RandomGenerator random = RandomManager.getRandom();
54  
55      ExecutorService exec = ExecutorUtils.buildExecutor("KSKETCH");
56      try {
57        for (int iter = 0; iter < cluster.getSketchIterations(); iter++) {
58          log.info("Starting sketch iteration {}", iter + 1);
59          List<Future<Collection<RealVector>>> futures = new ArrayList<>();
60          for (int foldId = 0; foldId < foldVecs.size(); foldId++) {
61            futures.add(exec.submit(
62                new SamplingRun(index, random, foldId, foldVecs.get(foldId), pointsPerIteration)));
63          }
64          // At the end of each iteration, gather up the sampled points to add to the index
65          List<Collection<RealVector>> newSamples = ExecutorUtils.getResults(futures);
66          for (int foldId = 0; foldId < foldVecs.size(); foldId++) {
67            for (RealVector v : newSamples.get(foldId)) {
68              index.add(v, foldId);
69            }
70          }
71          index.rebuildIndices();
72        }
73  
74        List<Future<List<WeightedRealVector>>> ret = new ArrayList<>();
75        for (int foldId = 0; foldId < foldVecs.size(); foldId++) {
76          ret.add(exec.submit(new AssignmentRun(index, foldId, foldVecs.get(foldId))));
77        }
78        return ExecutorUtils.getResults(ret);
79      } finally {
80        ExecutorUtils.shutdownNowAndAwait(exec);
81      }
82    }
83  
84    private static KSketchIndex buildIndex(List<List<RealVector>> foldVecs, ClusterSettings settings) {
85      if (foldVecs.isEmpty()) {
86        throw new IllegalStateException("No input vectors found for sketch building");
87      }
88      KSketchIndex index = new KSketchIndex(foldVecs.size(), foldVecs.get(0).get(0).getDimension(),
89          settings.getIndexBits(), settings.getIndexSamples(), 1729L);
90      for (int i = 0; i < foldVecs.size(); i++) {
91        if (!foldVecs.get(i).isEmpty()) {
92          index.add(foldVecs.get(i).get(0), i);
93        }
94      }
95      index.rebuildIndices();
96      return index;
97    }
98  }
